Module: myimg.utils.scalebar

This module defines function insert_scalebar, with the following features:

  • The function inserts a scalebar into a micrograph (or diffractogram).
  • The function employs many auxiliary functions defined this module.
  • The function is usually not called directly, but through myimg.api:
>>> # Import of MyImg + open some image
>>> # (the next command is the standard import of MyImg package
>>> import myimage.api as mi     
>>> img = mi.MyImage('some.png')
>>> # Insert scalebar + save the result
>>> # (the next command employs myimg.utils.scalebar.insert_scalebar function
>>> img.scalebar('rwi,100um')
>>> img.save_with_ext('_clm.png')

 39def insert_scalebar(my_img, pixsize=None, F=None, **kwargs):
 40    # Objective: To insert scalebar in a micrograph.
 41    # Note: this function is usually called by MyImg.scalebar method.
 42    # => see the original myimg.api.MyImg.scalebar method for more info.
 43    #-----
 44    
 45    # (0) Print messages if requested
 46    messages = kwargs.get('messages') or False
 47    if messages:
 48        print('Insert scalebar, start of function:')
 49        print(f'  Image  : {my_img.name}')
 50        print(f'  Pixels : {my_img.img.width}x{my_img.img.height}')
 51    
 52    # (1) Determine pixelsize.
 53    # => analyze the {pixelsize} argument
 54    if pixsize is None:
 55        # No pixelsize argument was given.
 56        # We assume that my_image was calibrated
 57        # and comes with {my_img.pixelsize} attribute.
 58        if my_img.pixsize is not None:
 59            pixel_size = my_img.pixsize
 60        else:
 61            print('Error when inserting scalebar!')
 62            raise(ValueError('Pixel size has not been defined.'))
 63    else:
 64        pixel_size = get_pixel_size(my_img, pixsize)
 65    # Save info about pixelsize back to my_img object
 66    # (this is important for possible further processing, such as FFT
 67    my_img.pixsize = pixel_size
 68    # Print info if requested
 69    if messages:
 70        print('Determined pixel size:')
 71        print(f'  Pixsize argument : {pixsize}')
 72        print(f'  Final pixel size : {pixel_size}')
 73
 74    # (2) Read multiplicative factor F
 75    # If F is not given, set it to 1 and ...
 76    #  in the following code multiply all relevant dimensions by F.
 77    # We CANNOT multiply directly myimg.settings.Scalebar variables
 78    #  Reason is a bit tricky: In scripts processing multiple images...
 79    #  ..the 1st image scalebar dimensions would be multiplied by F
 80    #  ..the 2nd image scalebar dimensions would be multiplied again => F*F
 81    if F is None:
 82        F = 1
 83    if messages:
 84        print(f'Multiplicative factor F : {F}')
 85    
 86    # (3) Initialize Scalebar = ScaleWithUnits object
 87    # (2a) Get lenght of scalebar (number + units)
 88    # (length is given as argument OR we calculate from img -size and pix-size
 89    length_arg = kwargs.get('length')
 90    if length_arg != None:
 91        length_of_scalebar = NumberWithUnits(length_arg)
 92    else:
 93        length_of_scalebar = calculate_scalebar_length(my_img, pixel_size, F)
 94    # (2b) Get length of scalebar in pixels
 95    length_in_pixels = \
 96        calculate_scalebar_length_in_pixels(length_of_scalebar, pixel_size)
 97    # (2c) Initialize scalebar = ScaleWithUnits object based on (2a, 2b, 2c)
 98    swu = ScaleWithUnits(
 99        number = length_of_scalebar.number,
100        units  = length_of_scalebar.units,
101        pixels = length_in_pixels)
102    # (2d) Adjust scalebar lenght
103    if length_arg != None:
104        # If lenght argument was given, just round the lenght-in-pixels.
105        swu.pixels = round(swu.pixels)
106    else:
107        # If length was calculated, adjust the size to a reasonable value.
108        swu.adjust_scalebar_size()
109    # (2e) Print info if requested
110    if messages:
111        print('Calculated scalebar size:')
112        print(f'  {swu}')
113    
114    # (3) Collect additional keyword arguments.
115    # (3a) key arguments are collected here
116    position = kwargs.get('position') or Settings.Scalebar.position
117    color    = kwargs.get('color')    or Settings.Scalebar.color
118    bcolor   = kwargs.get('bcolor')   or Settings.Scalebar.bcolor
119    # (3b) Special case - tranparent background: bcolor='transparent' => None
120    # (bcolor=None would change bcolor to default = Settings.Label.bcolor
121    if bcolor == 'transparent': bcolor = None
122    # (3c) Additional arguments/parameters are collected in Settings.Scalebar
123    # (they can be modified in code: Settings.Scalebar.font = 'arial.ttf'
124    # (some of them are are read below for the sake of convenience    
125    
126    # (4) Calculate font size + prepare font
127    font_name = Settings.Scalebar.font
128    text_height_rel = Settings.Scalebar.text_height * F
129    text_height_pix = round(text_height_rel * my_img.img.height) 
130    font_size = my_img.set_font_size(font_name, text_height_pix)
131    font_object = ImageFont.truetype(font_name, font_size)
132    ImageDraw.fontmode = 'L'
133    draw = ImageDraw.Draw(my_img.img)
134    bbox = draw.textbbox((20, 20), swu.text(), font=font_object)
135    scalebar_text_size = (bbox[2]-bbox[0], bbox[3]-bbox[1])
136    if messages:
137        print('Collected & calculated font parameters:')
138        print(f'  Font name             : {font_name}')
139        print(f'  Multiplicative factor : {F}')
140        print(f'  Text_height_rel       : {text_height_rel}')
141        print(f'  Text_height_pix       : {text_height_pix}')
142        print(f'  Font_size             : {font_size}')
143        print(f'  Scalebar text size    : {scalebar_text_size}')
144            
145    # (5) Calculate position and shape of the scalebar
146    # (scalebar consists of: text and bar/line and box/background
147    # (5a) Dimensions of text, bar and box 
148    # ... xy-size of scalebar text (already calculated above)
149    text_x, text_y = scalebar_text_size
150    # ... xy-size of bar = scalebar line
151    bar_x = swu.pixels
152    bar_y = my_img.img.height * Settings.Scalebar.line_height * F
153    # ... xy-size of box = scalebar background
154    box_x = max(text_x,bar_x) + \
155        my_img.img.height * (2 * Settings.Scalebar.edge_x * F) 
156    box_y = text_y + my_img.img.height * (
157        2 * Settings.Scalebar.edge_y * F +
158        Settings.Scalebar.line_height * F + 
159        Settings.Scalebar.separator * F)
160    # (5b) (x1,y1) = position of the upper left corner of the box
161    if position != None:
162        x1 = position[0]
163        y1 = position[1]
164    else:
165        # x1,y1 = position of the upper left corner of the box
166        x1 = my_img.img.height * Settings.Scalebar.offset_x * F + box_x
167        y1 = my_img.img.height * Settings.Scalebar.offset_y * F + box_y
168        x1 = round(my_img.img.width  - x1)
169        y1 = round(my_img.img.height - y1)
170    # (5c) (x2,y2) position of the upper left corner of the bar/line
171    x2 = round(x1 + my_img.img.height * Settings.Scalebar.edge_x * F)
172    y2 = round(y1 + box_y - my_img.img.height * (
173        Settings.Scalebar.edge_y * F +
174        Settings.Scalebar.line_height * F)) 
175    if text_x > bar_x:
176        x2 = x2 + round((text_x - bar_x)/2)
177    # (5d) (x3,y3) position of the text
178    # (XY-anchor of the text will be 'mb' = middle,bottom
179    # (x3 => anchor 'm' => in the middle of the box
180    # (y3 => anchor 'b' => y-position of the bar/line - separator
181    x3 = round(np.mean([x1, x1 + box_x]))
182    y3 = round(y2 - my_img.img.height * Settings.Scalebar.separator * F)
183    # (5e) Print info if requested
184    if messages:
185        print('Calculated scalebar geometry:')
186        print('  Textsize in pixels       :', text_x, text_y)
187        print('  Box, upper left corner   :', x1, y1)
188        print('  Line, upper left corner  :', x2, y2)
189        print('  Text, center (anchor=mb) :', x3, y3)
190    
191    # (6) Draw scalebar in image
192    # Draw box/background 
193    if bcolor != None: draw.rectangle([x1,y1,x1+box_x,y1+box_y], fill=bcolor)
194    # Draw line
195    draw.rectangle([x2,y2,x2+bar_x,y2+bar_y], fill=color)
196    # Draw text
197    draw.text((x3,y3), swu.text(), font=font_object, fill=color, anchor='mb')
198    # Print final info if requested
199    if messages:
200        print('End of function, the scalebar drawn in the image object.')
201    
202    # (7) End of function.
203    # (No return value; the scalebar is drawn in my_img.img object

206def get_pixel_size(my_img, pixsize):
207    # Objective: to get pixel size from pixsize argument + my_img object props.
208    #-----
209    
210    # (1) Split pixelsize argument to components
211    pixelsize_list = pixsize.split(',')
212    pixelsize_from = pixelsize_list[0]
213    pixelsize_args = pixelsize_list[1:]
214    # (2) Decide how to determine pixel size value + call correct funtion 
215    if   pixelsize_from == 'rwi':
216        pixel_size = pixel_size_from_rwi(my_img, pixelsize_args)
217    elif pixelsize_from == 'knl':
218        pixel_size = pixel_size_from_knl(my_img, pixelsize_args)
219    elif pixelsize_from == 'mag':
220        pixel_size = pixel_size_from_mag(my_img, pixelsize_args)
221    elif pixelsize_from == 'txt':
222        pixel_size = pixel_size_from_txt(my_img, pixelsize_args)
223    else:
224        print('Unkwnown pixsize argument:', pixsize)
225    # (3) Return final pixel size
226    # (pixel_size type => myimg.utils.scalebar.NumberWithUnits
227    return(pixel_size)

230def pixel_size_from_rwi(my_img, pixsize_args):
231    # Objective: to get pixel size from RWI + image_width.
232    #-----
233    
234    # (1) RWI = the 1st pixsize argument after rwi (example: 'rwi,100um')
235    rwi   = NumberWithUnits(pixsize_args[0])
236    # (2) Pixel size = rwi / image_width_in_pixels (units = rwi.units)
237    pixel = NumberWithUnits(
238        number = rwi.number/my_img.img.width,
239        units  = rwi.units)
240    # (3) Return pixel_size (as pixel variable with type: NumberWithUnits)
241    return(pixel)

246    # (my_img is just a formal argument here

260    pixel = NumberWithUnits(

262        units  = knl.units)
263    # (3) Return pixel_size (as pixel variable with type: NumberWithUnits)
264    return(pixel)

267def pixel_size_from_mag(my_img, pixelsize_args):
268    # Objective: to get pixel size from pixsize='mag...' argument.
269    # Note: this works only for the calibrated microscopes
270    # (calibrated microscope = a dataclass in myimg.settings.MicCalibrations
271    #-----
272    
273    # (1) The first argument should be calibrated microscope name
274    try:
275        calibrated_microscope = getattr(
276            Settings.MicCalibrations, pixelsize_args[0])
277    except AttributeError:
278        print('Uknown microscope in pixsize argument!')
279        print('Pixsize argument:', pixelsize_args)
280        sys.exit()
281    
282    # (2) Get magnification...
283    # (2a) If the 2nd argument was given, it should be the magnification
284    if len(pixelsize_args) == 2:
285        mag = pixelsize_args[1]
286    # (2b) If 2nd argument was not given, get it from image name
287    else:
288        # Step 1: Get image name, my_img.name can be str or Path!
289        if isinstance(my_img.name, str):
290            image_name = my_img.name
291        elif isinstance(my_img.name, Path):
292            image_name = str(my_img.name)
293        else:
294            raise('Unknown type in MyImage.name!')
295        # Step 2: Get magnification from the image name
296        mag = re.search(r'\S+_(\d+\.?\d*[Kk]?[Xx]?)\.\S{3,}', image_name)[1]
297        if mag == None:
298            print('Unrecognized magnification in filename!')
299            print('Filename:', my_img.name)
300            sys.exit()
301    # (2c) Convert the magnification to number
302    # Replace final 'x' with the empty string
303    if mag.endswith('x'):
304        mag = mag.replace('x','')
305    # Replace final 'k' with the three zeros (if it is there)
306    if mag.endswith('k'):
307        mag = mag.replace('k','000')
308    # Convert magnification to float
309    try:
310        mag = float(mag)
311    except ValueError:
312        print('Unclear magnification in pixsize argument!')
313        print('Pixsize argument:', pixelsize_args)
314        sys.exit()
315        
316    # (3) Now we should have calibrated_microscope name + magnification
317    # Get calibration constant with units
318    const = calibrated_microscope.const
319    units = calibrated_microscope.units
320    # Calculate RWI = real-width-of-image (simple relation: rwi = const/mag) 
321    rwi = NumberWithUnits(
322        number = const/mag,
323        units  = units)
324    # Set correct units for RWI
325    # (this is not strictly necessary, but it is more correct
326    rwi.set_correct_units()
327    # Recalculate RWI to pixel_size
328    pixel = NumberWithUnits(
329        number = rwi.number/my_img.img.width,
330        units  = rwi.units)
331    
332    # (4) Return pixel_size (as pixel variable with type: NumberWithUnits)
333    return(pixel)

336def pixel_size_from_txt(my_img, pixelsize_args):
337    # Objective: to get pixel size from  pixsize='txt...' argument.
338    # Note: this works only for microscopes with known description files.
339    # (known descr.file = a dataclass in myimg.settings.MicDescriptionFiles
340    #-----
341    
342    # (1) Get the name of mic_description_file class from pixsize_args
343    # Example:
344    #  - we have a string: pixsize_args='MAIA' 
345    #  - and we need an object name: Settings.MicDescriptionFiles.MAIA
346    # => GoogleSearch: python access variable/object/class when I know string
347    # => https://stackoverflow.com/q/9437726
348    # => https://stackoverflow.com/q/1167398
349    microscope = pixelsize_args[0]
350    mic_description_file = getattr(
351        Settings.MicDescriptionFiles, microscope)
352
353    # (2) Get the text description file name 
354    # once we know the mic_description_file class from the previous step
355    # Example:
356    #  - we have: my_img object + mic_description_class object
357    #  - and we need text description file name, such as: 'FIGS\img123.hdr'
358    img_filename_without_extension = os.path.splitext(my_img.name)[0]
359    text_description_filename_general = mic_description_file.filename
360    text_description_filename = \
361        text_description_filename_general.replace(
362            '*', img_filename_without_extension)
363    # Yet final conversion - some description files can contain wildcards
364    # Example: Settings.MicDescriptionFiles.MAIA.filename = '*-???.hdr'
365    # Trick to solve this: convert wildcards to filename using glob function
366    text_description_filename = glob.glob(text_description_filename)[0]
367    
368    # (3) Open the text description file and get pixel size
369    # (a) Prepare varialbles
370    pixsize_regexp = re.compile(mic_description_file.pixsize_line)
371    pixsize_units  = mic_description_file.pixsize_units
372    pixsize_number = None
373    # (b) Search for pixel size in the text description file
374    # (considering all possible exceptions, as specified below
375    try:
376        fh = open(text_description_filename)
377        for line in fh:
378            m = pixsize_regexp.search(line)
379            if m:
380                pixsize_number = float(m.group(1))
381                break
382    # (c) Considering all possible exceptions ...
383    except FileNotFoundError:
384        print('Error when trying to read image description file!')
385        print(f'File [{text_description_filename}] not found.')
386        sys.exit()
387    except Exception as err:
388        print('Error when trying to read image description file!')
389        print('Exception name:', type(err).__name__)
390        print('Description:', err)
391        sys.exit()
392
393    # (4) Create NumberWithUnits object
394    # (If pixel size was not found => pixsize_number = None
395    # (exception will be raised automatically - do do not have to worry.
396    # (The exception is treated directly in NumberWithUnits class.
397    pixel = NumberWithUnits(
398        number = pixsize_number,
399        units  = pixsize_units)
400
401    # (5) Return pixel_size (as pixel variable with type: NumberWithUnits)
402    return(pixel)

405def calculate_scalebar_length(my_img, pixel_size, F):
406    # Objective: to calculate suitable scalebar length,
407    # based on image size, pixel size, Settings.Scalebar.length and factor F.
408    # (Note: we calculate just length+units;
409    # ( lenght-in-pixels is calcualted in the next step
410    # (Reason: step-by-step approach - we will do this in the next function;
411    # ( for length-in-pixels we need the same units of scalebar and pixel_size
412    #-----
413    
414    # (1) Determine RWI from image size and pixel size.
415    rwi = NumberWithUnits(
416        number = my_img.img.width * pixel_size.number,
417        units  = pixel_size.units)
418    
419    # (2) Determine suitable lenght of scalebar
420    # (At the moment, we determine just number+units
421    # (  Lenght-in-pixels will be determined later
422    # (  Reason: for lenght-in-pixels we have to consider pixel-size units
423    nwu = NumberWithUnits(
424        # Scalebar length should be a fraction of image width
425        # This fraction of image width is saved in Settings.Scalebar.lenght
426        # Moreover, if multiplicative coeff F was given, is should be Fx more.
427        number = rwi.number * Settings.Scalebar.length * F,
428        units  = rwi.units)
429    return(nwu)

432def calculate_scalebar_length_in_pixels(length_of_scalebar, pixel_size):
433    # Objective: to calculate scalebar length-in-pixels
434    # (using two NumberWithUnits objects: length_of_scalebar + pixel_size
435    #-----   
436    # (1) pixel_size.units should be the same as length_of_scalebar.units
437    pixel_size.set_units_to(length_of_scalebar.units)
438    # (2) If units are the same, the calculation is easy
439    # Example: scalebar = 5um, pixel_size = 0.1um => length_in_pix = 5/0.1 = 50 pix
440    length_in_pixels = length_of_scalebar.number / pixel_size.number
441    # (3) Return calculated length_of_scalebar_in_pixels
442    return(length_in_pixels)

195    # Draw line
196    draw.rectangle([x2,y2,x2+bar_x,y2+bar_y], fill=color)
197    # Draw text
198    draw.text((x3,y3), swu.text(), font=font_object, fill=color, anchor='mb')
199    # Print final info if requested
200    if messages:
201        print('End of function, the scalebar drawn in the image object.')
