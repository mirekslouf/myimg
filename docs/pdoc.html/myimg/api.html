myimg.api

Module: myimg.api

A simple interface to package myimg.

>>> # Simple usage of myimg.api interface
>>> import myimage.api as mi
>>>
>>> # (1) Open image
>>> img = mi.MyImage('somefile.bmp')  # input image: somefile.bmp
>>>
>>> # (2) Modify the image 
>>> img.cut(60)                # cut off lower bar (60 pixels)             
>>> img.label('a')             # label to the upper-left corner
>>> img.scalebar('rwi,100um')  # scalebar to the lower-right corner
>>>
>>> # (3) Save the modified image 
>>> img.save_with_ext('_clm.png')  # output: somefile_clm.png

 89    def get_font_size(self, font_name, required_font_size_in_pixels):
 90        '''
 91        Get font size (in fontsize units)
 92        corresponding to *required_font_size_in_pixels*.
 93
 94        Parameters
 95        ----------
 96        fontname : str
 97            Name of the TrueType font to use.
 98            Example (working in Windows): font_name='timesbd.ttf'
 99        required_font_size_in_pixels : float
100            Required font size in pixels.
101
102        Returns
103        -------
104        font_size : int
105            Final font size (in font units).
106            If the returned *font_size* (in fontsize units)
107            is applied to font with given *fontname*,
108            then the height of the font (in pixels units)
109            will correspond to *required_font_size_in_pixels* argument.
110            
111        Technical notes
112        ---------------
113        * This function is a modified recipe from StackOverflow:
114          https://stackoverflow.com/q/4902198
115        * My modification may be a bit slower (not too much)
116          but it seems to be clear and reliable.
117        '''
118        # (1) Initial fontsize = required_font_size
119        # (Note: fontsize [in fontsize units]
120        # (is just APPROXIMATELY equal to required fontsize [in pixels].
121        font_size = round(required_font_size_in_pixels)
122        
123        # (2) Initialize draw_object + font_object and calculate font height
124        # (Note1: draw_object is needed for the correct font height calculation
125        # (Note2: we calculate the font height for a model text - here: cap 'M'
126        draw_object = ImageDraw.Draw(self.img)
127        font_object = ImageFont.truetype(font_name, font_size)
128        font_height = MyImage.font_height_pix(draw_object, font_object) 
129        # (2a) Current font_height > required_font_size_in_pixels, decrease...
130        while font_height > required_font_size_in_pixels:
131            font_size -= 1
132            font_object = ImageFont.truetype(font_name, font_size)
133            font_height = MyImage.font_height_pix(draw_object, font_object)
134        # (2b) Current font_height > required_font_size_in_pixels, increase...
135        while font_height < required_font_size_in_pixels:
136            font_size += 1
137            font_object = ImageFont.truetype(font_name, font_size)
138            font_height = MyImage.font_height_pix(draw_object, font_object)
139        
140        # (3) Return font size (for given font_name, in font_name units)
141        return(font_size)

148        return(text_height)

247    def scalebar(self, pixsize, F=None, **kwargs):
248        '''
249        Insert a scalebar in the lower right corner of the image.
250
251        Parameters
252        ----------
253        pixsize : str
254            Description how to determine pixel size.
255            Pixel size is needed to calculate the scalebar length.
256            See *Example* section below to see available options.
257        F : float, optional, the default is None
258            Multiplication coefficient/factor that changes the scalebar size.
259            If F = 1.2, then all scalebar parameters are enlarged 1.2 times.
260        kwargs : list of keyword arguments
261            Allowed keyword arguments are:
262            color, bcolor, position, stripes, messages.
263            See section *List of allowed kwargs* for detailed descriptions. 
264            
265        Returns
266        -------
267        None
268            The scalebar is drawn directly to *self.img*.
269            
270        List of allowed kwargs
271        ----------------------
272        * color : PIL color specification, default is 'black'.
273            Text color = color of the scalebar text and line.
274        * bcolor : PIL color specification, default is 'white'.
275            Background color = color of the scalebar background/box.
276        * length : str, default is None.
277            If length is given (using a string such as '100um','1.2nm')
278            then the lenght fixed at given value and not calculated by the
279            program (calculation would yield some reasonable lenght of
280            scalebar around 1/6 of the image width; this default is saved
281            in myimg.settings.Scalebar.length property - which can be changed).
282        * position : list or tuple or None, default is None.
283            If position = None, the scalebar is drawn
284            at the default position in the lower-right corner of the image.
285            If position = (X,Y) = list or tuple of two integers,
286            the scalebar is drawn at position X,Y of the image.
287        * stripes : bool or int, default is False.
288            If stripes = False, draw standard scalebar.
289            If stripes = True or 1, draw scalebar with 5 stripes.
290            If stripes = N, where N>=2, draw striped scalebar with N stripes.
291        * messages : bool, default is False.
292            If messages=True, print info about program run.
293        
294        Example
295        -------
296        >>> # Four basic ways how to insert a scalebar in an image
297        >>> # (model example; in real life we use just one of the ways
298        >>>
299        >>> # (0) Import api + read image
300        >>> import myimage.api as mi
301        >>> img = mi.MyImage('../IMG/image123_20kx.bmp')
302        >>> 
303        >>> # (1) Pixel size from real width of image = 100um
304        >>> img.scalebar('rwi,100um')
305        >>>
306        >>> # (2) Pixel size from a known length in image => 50 nm = 202 pixels
307        >>> img.scalebar('knl,50nm,202')
308        >>> 
309        >>> # (3) Pixel size from known magnification
310        >>> # (note: this can be done only for calibrated microscope
311        >>> # (calibrated microscopes => myimg.settings.MicCalibrations
312        >>> # (3a) magnification deduced from last part of image name
313        >>> # (note: mag = everything between last underscore and suffix
314        >>> # (in this example we have: ../IMG/image123_20kx.bmp => mag = 20kx
315        >>> img.scalebar('mag,TecnaiVeleta')
316        >>> # (3b) magnification inserted directly
317        >>> # (note: mag can be something like 20kx, 20k, 20000x, 20000
318        >>> img.scalebar('mag,TecnaiVeleta,20kx')
319        >>>
320        >>> # (4) Pixel size from accompanying text file
321        >>> # (note: some microscopes save images + descriptive txt files
322        >>> # (the format of text file must be described somehow
323        >>> # (description of text files => myimg.settings.MicDescriptionFiles
324        >>> img.scalebar('txt,MAIA3')
325        >>> 
326
327        Technical notes
328        ---------------
329        * Transparent background:
330          To set transparent background,
331          set optional/keyword argument bcolor='transparent'.
332          It is not enough to omit bcolor,
333          because all omitted keyword arguments
334          are set to their defaults defined in Settings.Scalebar.
335          In the case of omitted bcolor argument, the default is 'white'. 
336        * Color scalebar in grayscale image:
337          To set color label in grayscale image,
338          it is necessary to convert image to RGB;
339          otherwise the colored label would be converted to grayscale.
340        '''
341        
342        # The complete code of this method is long. 
343        # Therefore, the code has been moved to its own module.
344        # This method is a wrapper calling the function in the external module. 
345        from myimg.utils import scalebar as my_scalebar
346        my_scalebar.insert_scalebar(self, pixsize, F, **kwargs)

 91        '''
 92        Get font size (in fontsize units)
 93        corresponding to *required_font_size_in_pixels*.
 94
 95        Parameters
 96        ----------
 97        fontname : str
 98            Name of the TrueType font to use.
 99            Example (working in Windows): font_name='timesbd.ttf'
100        required_font_size_in_pixels : float
101            Required font size in pixels.

104        -------

107            If the returned *font_size* (in fontsize units)

109            then the height of the font (in pixels units)
110            will correspond to *required_font_size_in_pixels* argument.
111            
112        Technical notes
113        ---------------
114        * This function is a modified recipe from StackOverflow:
115          https://stackoverflow.com/q/4902198
116        * My modification may be a bit slower (not too much)
117          but it seems to be clear and reliable.
118        '''
119        # (1) Initial fontsize = required_font_size
120        # (Note: fontsize [in fontsize units]
121        # (is just APPROXIMATELY equal to required fontsize [in pixels].
122        font_size = round(required_font_size_in_pixels)
123        
124        # (2) Initialize draw_object + font_object and calculate font height
125        # (Note1: draw_object is needed for the correct font height calculation
126        # (Note2: we calculate the font height for a model text - here: cap 'M'
127        draw_object = ImageDraw.Draw(self.img)
128        font_object = ImageFont.truetype(font_name, font_size)
129        font_height = MyImage.font_height_pix(draw_object, font_object) 
130        # (2a) Current font_height > required_font_size_in_pixels, decrease...
131        while font_height > required_font_size_in_pixels:
132            font_size -= 1
133            font_object = ImageFont.truetype(font_name, font_size)
134            font_height = MyImage.font_height_pix(draw_object, font_object)
135        # (2b) Current font_height > required_font_size_in_pixels, increase...
136        while font_height < required_font_size_in_pixels:
137            font_size += 1
138            font_object = ImageFont.truetype(font_name, font_size)
139            font_height = MyImage.font_height_pix(draw_object, font_object)
140        
141        # (3) Return font size (for given font_name, in font_name units)
142        return(font_size)

 51    '''
 52    Microscope calibration constants.
 53    
 54    * This dataclass is a container for several sub-dataclasses.
 55    * The subdataclasses define the individual calibrated microscopes.
 56    
 57    Calculation of calibration constant for given microscope
 58    
 59    * mag   = magnification, for which we know rwi
 60    * rwi   = real width of image (in our case given in [mm]
 61    * const = calibration constant = rwi * mag
 62    
 63    Usage of calibration constants
 64    
 65    * rwi at given mag: rwi = const/mag (const = calibration constant)
 66    * Warning: for EM microscopes, these relations may be just approximate
 67    
 68    Justification
 69    
 70    * 2x higher mag => 2x lower rwi
 71    * 4x higher mag => 4x lower rwi
 72    * Therefore (according to elementary logic)
 73        - rwi = some_constant/mag
 74        - const = some_constant = calibration_constant = rwi * mag
 75        - physical meaning of calibration const:
