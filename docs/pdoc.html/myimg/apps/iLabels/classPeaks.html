109    def show_in_image(self):
110        '''
111        Display the image with the peak data overlay (if image and data exist),
112        with different colors for different particle types.
113    
114        Parameters
115        ----------
116        self : object
117            The instance of the class containing the image and peak data.
118    
119        Returns
120        -------
121        None
122            The method does not return a value but displays an image with
123                overlayed peak data.
124    
125        Raises
126        ------
127        ValueError
128            If the DataFrame does not contain the required columns.
129        '''
130        if self.img is None:
131            print("No image to display.")
132            return
133        if self.df.empty:
134            print("No peak data to overlay on the image.")
135            return
136    
137        # Check if the DataFrame contains the required columns
138        if 'X' not in self.df.columns or 'Y' not in self.df.columns:
139            print("Peak data does not contain 'X' and 'Y' columns.")
140            return
141        if 'Class' not in self.df.columns:
142            print("Peak data does not contain 'Class' column.")
143            return
144    
145        # Define a dictionary mapping particle types to colors
146        color_map = {
147            '1': 'red',
148            '2': 'blue',
149            '3': 'green',
150            '4': 'purple',
151        }
152    
153        # Plot the image
154        plt.imshow(self.img)
155    
156        # Loop through each unique particle type,
157        # and plot the peaks with the corresponding color
158        for particle_type in self.df['Class'].unique():
159            particle_data = self.df[self.df['Class'] == particle_type]
160            plt.scatter(particle_data['X'], particle_data['Y'], 
161                        c=color_map.get(str(particle_type), 'black'),
162                        # Default to black if type is not in the map
163                        label=particle_type, 
164                        s=25, marker='+')
165        
166        plt.legend(title="Particle Type", loc='center left', bbox_to_anchor=(1.05, 0.5))
167        plt.axis("off")
168        plt.title(f"Peaks on {self.img_name}")
169        plt.show()

208            from myimg.utils.iplot import interactive_plot, default_plot_params

210            # Generate and display the interactive plot for manual annotation
211            fig, ax = interactive_plot(self.img, 
212                                       default_plot_params(self.img),
213                                       filename=self.file_name, 
214                                       messages=self.messages)
215            plt.show()
216            
217    
218        elif method == "ccorr":
219            # Load masks

229            self.detected = milab.detectors.detector_correlation(self.img,

245    def characterize(self, img_path, peak_path, mask_path, 
246                     imID='im0x', preprocess=True, show=False):
247        """
248        Characterizes image peaks by extracting ROIs, computing features, and 
249        selecting the most informative ones for classification.
250        
251        Parameters
252        ----------
253        img_path : str
254            Path to the image file (e.g., '.tif') to analyze.
255        
256        peak_path : str
257            Path to the pickled file containing detected peak coordinates 
258            (typically a DataFrame or array).
259        
260        mask_path : str
261            Directory containing class masks as 'mask1.pkl' to 'mask4.pkl'.
262        
263        imID : str, optional
264            Identifier for the image, used for internal labeling
265            Default is 'im0x'
266        
267        preprocess : bool, optional
268            Whether to preprocess the input image (CLAHE, gamma correction, 
269            normalization). Default is True.
270        
271        show : bool, optional
272            If True, displays example ROIs and feature visualizations.
273            Dfault is False.
274        
275        Returns
276        -------
277        None
278            This method modifies the internal state of the object by setting 
279            attributes such as self.features, self.selection, self.X_train, 
280            self.X_test, etc.
281        """
282
283        # (1) Load masks
284        self.masks = {}
285        for i in range(1, 5):
286            file_path = os.path.join(mask_path, f"mask{i}.pkl")
287            if not os.path.exists(file_path):
288                raise FileNotFoundError(f"Input file not found: {file_path}")
289            with open(file_path, 'rb') as f:
290                self.masks[i] = pickle.load(f)
291        
292        # Calculate ROI shape based on the mask shape
293        s = int(self.masks[i].shape[0]/2)
294        
295        # (2) Preprocess image
296        if preprocess:
297           self.pimg =  milab.roi.preprocess_image(self.img, 
298                                               apply_clahe=True, 
299                                               gamma=1.2, 
300                                               normalize=True)
301        
302        # (3) Prepare data for ROI extraction
303        self.arr, self.df, _  = milab.roi.prep_data(img_path, peak_path, 
304                                             min_xy=20, imID=imID, show=show)
305        
306        # (4) Extract ROIs from image
307        self.rois, self.dfs = [], []
308        rois, df = milab.roi.get_ROIs(im=self.pimg, 
309                                            df=self.df, 
310                                            s=s, 
311                                            norm=False, 
312                                            show=show)
313        
314
315        self.dfs.append(df)        
316        self.rois = [(roi, imID) for roi in rois]        
317        self.dfs = pd.concat(self.dfs, ignore_index=True)
318        
319        
320        # Show examples of extracted roi if show=True
321        if show:
322            milab.roi.show_random_rois(self.rois, self.dfs, n=5)
323            
324        
325        # (5) Calculate features
326        self.features, _ = milab.features.get_features(self.rois, self.dfs, 
327                                                       self.masks,
328                                                       show=False)
329        
330        self.features = self.features.dropna()
331        
332        if show:
333            milab.features.visualize_features(self.features, method="box")
334        milab.features.visualize_features(self.features, method="heat")
335        
336        # (6) Select features
337        # Split dataset
338        self.X_train, self.X_test, self.y_train, self.y_test = \
339            milab.classifiers.dataset(self.features)
340        
341        # Select features
342        self.selection = milab.classifiers.select_features(self.X_train, 
343                                                           self.y_train, 
344                                                           num=5, 
345                                                           estimator=None)
346        
347        return

350    def correct(image, coords, s=20, method='intensity'):
351        """
352        Refines or corrects peak coordinates based on a specified method.
353        
354        Parameters
355        ----------
356        image : np.ndarray
357            The input image in which the coordinates will be corrected
358        
359        coords : pandas.DataFrame
360            DataFrame containing peak coordinates (e.g., 'X' and 'Y' columns).
361        
362        s : int, optional
363            Half-size of the square ROI to extract around each coordinate. 
364            Necessary for "intensity" method. Default is 20.
365        
366        method : str, optional
367            Correction method to use. Currently only 'intensity' is supported 
368            (uses intensity-based ROI extraction). Default is 'intensity'.
369        
370        Returns
371        -------
372        pandas.DataFrame
373            Updated coordinates DataFrame after correction.
374        """
375
376        if method == "intensity":
377           _, peaks =  milab.roi.get_ROIs(im=image, df=coords, 
378                                      s=s, norm=False, show=False)
379        else: 
380            raise ValueError("Currently, only the 'intensity'\
381                             method is supported.")
382           
383        return peaks

386    def classify(self, data, method='gauss_fit', target=None, estimator=None, param_dist=None, sfeatures=None):
387        """
388        Classifies input data using the specified classification method.
389        
390        Parameters
391        ----------
392        data : pandas.DataFrame or np.ndarray
393            The feature data to classify.
394        
395        method : str, optional
396            The classification method to use. Currently, only 'rfc' (Random 
397            Forest Classifier) is supported. 
398            Default is 'gauss_fit' (placeholder for future extensions).
399        
400        target : array-like, optional
401            Ground truth labels for the data (used for evaluation if provided).
402        
403        estimator : RandomForestClassifier or str, optional
404            A pre-trained classifier or a path to a saved estimator file (.pkl).
405            If None, a new RandomForestClassifier will be optimized and trained.
406        
407        param_dist : dict, optional
408            Hyperparameter search space for optimizing the RandomForestClassifier.
409            Used only when training a new estimator.
410        
411        sfeatures : list of str, optional
412            List of selected feature names to use for classification. If None, 
413            uses `self.selection`.
414        
415        Returns
416        -------
417        y_pred : np.ndarray
418            Predicted class labels for the input data.
419        
420        Raises
421        ------
422        ValueError
423            If the provided estimator is neither a valid path nor a classifier 
424            instance.
425            
426        """
427
428        if method == "rfc":
429            # If no estimator is provided, optimize and train a new Random Forest
430            if estimator is None:
431                # Perform hyperparameter search and get an optimized classifier
432                rfc_opt = milab.classifiers.get_optimal_rfc(self.X_train,
433                                                  self.y_train, 
434                                                  param_dist)
435                
436                # Fit the optimized classifier to the training data
437                self.rfc_fit, _ = milab.classifiers.fitting(self.X_train, 
438                                                  self.y_train, 
439                                                  estimator=rfc_opt, 
440                                                  reports=True, 
441                                                  sfeatures=self.selection)
442                
443                # Predict class labels on the input data using the classifier
444                self.y_pred = milab.classifiers.predicting(data, 
445                                                 estimator=self.rfc_fit, 
446                                                 sfeatures=self.selection, 
447                                                 y_test=target)
448            else: 
449                # Load the pre-trained classifier if estimator is a file path
450                if isinstance(estimator, str):
451                    estimator = joblib.load(estimator)
452
453                elif not isinstance(estimator, ClassifierMixin):
454                    # Raise error if estimator is not a valid classifier object
455                    raise ValueError("Provided estimator is invalid.")
456                
457                # Use default selected features if none are provided
458                if sfeatures is None:
459                    sfeatures = self.selection
460                
461                # Predict class labels using the provided classifier
462                self.y_pred = milab.classifiers.predicting(data, 
463                                                 estimator, 
464                                                 sfeatures, 
465                                                 y_test=target)
466                
467        return self.y_pred
