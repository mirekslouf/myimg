152    def show_in_image(self, cmap="viridis"):
153        """
154        Overlay peak locations on the image, colored by class.
155    
156        Plots ``self.img`` and scatters the peak coordinates from ``self.df``.
157        Points are colored by the value in the ``'Class'`` column.
158    
159        Parameters
160        ----------
161        cmap : str, optional
162            Matplotlib colormap used when displaying single-channel (grayscale)
163            images. Ignored for RGB/RGBA images. Default is ``"viridis"``.
164    
165        """
166        if self.img is None:
167            print("No image to display.")
168            return
169        if self.df.empty:
170            print("No peak data to overlay on the image.")
171            return
172    
173        # Check if the DataFrame contains the required columns
174        if 'X' not in self.df.columns or 'Y' not in self.df.columns:
175            print("Peak data does not contain 'X' and 'Y' columns.")
176            return
177        if 'Class' not in self.df.columns:
178            print("Peak data does not contain 'Class' column.")
179            return
180    
181        # Define a dictionary mapping particle types to colors
182        color_map = {
183            '1': 'red',
184            '2': 'blue',
185            '3': 'green',
186            '4': 'purple',
187        }
188    
189        # Plot the image
190        plt.imshow(self.img, cmap=cmap)
191    
192        # Loop through each unique particle type,
193        # and plot the peaks with the corresponding color
194        for particle_type in self.df['Class'].unique():
195            particle_data = self.df[self.df['Class'] == particle_type]
196            plt.scatter(particle_data['X'], particle_data['Y'], 
197                        c=color_map.get(str(particle_type), 'black'),
198                        # Default to black if type is not in the map
199                        label=particle_type, 
200                        s=25, marker='+')
201        
202        plt.legend(title="Particle Type", 
203                   loc='center left', 
204                   bbox_to_anchor=(1.05, 0.5))
205        plt.axis("off")
206        plt.show()

209    def find(self, 
210             method='manual', 
211             ref=True, 
212             mask_path=None, 
213             midx=0, 
214             thr=0.5, 
215             show=True, 
216             **kwargs):
217        """
218        Find peaks either interactively (manual) or by norm. cross-correlation
219        (ccorr) against a reference mask.
220    
221        Parameters
222        ----------
223        method : {"manual", "ccorr"}, optional
224            Detection mode. "manual" opens an interactive picker; "ccorr" 
225            performs normalized cross-correlation with a reference mask. 
226            Default "manual".
227            
228        ref : bool, optional
229            Whether to refine detected coordinates after the initial pick 
230            (reserved for future use). Default True.
231            
232        mask_path : str, optional
233            Directory containing reference masks saved as ``mask1.pkl``,
234            ``mask2.pkl``,... Required when ``method="ccorr"``.
235            
236        midx : int, optional
237            Mask index to use. Default 0.
238            
239        thr : float, optional
240            NCC threshold in [0, 1]. Detections with correlation below this are
241            discarded. Default 0.5.
242            
243        show : bool, optional
244            Visualize the detections/overlay. Default True.
245            
246        **kwargs
247            Reserved for passing additional options to the interactive plotter
248            or detector implementation (e.g., debug flags).
249    
250        Returns
251        -------
252        detected : Any or None
253            For ``method="ccorr"``, returns whatever 
254            ``milab.detectors.detector_correlation`` returns (e.g., a DataFrame 
255            of coordinates/scores). For ``method="manual"``, returns ``None`` 
256            after displaying the picker.
257
258        """
259    
260        if method == "manual":
261            from myimg.utils.iplot import interactive_plot, default_plot_params
262    
263            # Generate and display the interactive plot for manual annotation
264            fig, ax = interactive_plot(self.img, 
265                                       default_plot_params(self.img),
266                                       filename=self.file_name, 
267                                       messages=self.messages)
268            plt.show()
269            
270    
271        elif method == "ccorr":
272            # Load masks
273            self.masks = {}
274            for i in range(1, 5):
275                file_path = os.path.join(mask_path, f"mask{i}.pkl")
276                if not os.path.exists(file_path):
277                    raise FileNotFoundError(
278                        f"Input file not found: {file_path}")
279                with open(file_path, 'rb') as f:
280                    self.masks[i] = pickle.load(f)
281
282                    
283            # Proceed with detector-based correlation using the mask
284            self.detected = milab.detectors.detector_correlation(self.img, 
285                                                     self.masks[midx], 
286                                                     thr, 
287                                                     show)
288            
289            return self.detected
290    
291        else:
292            raise ValueError("Invalid detection method. Use 'manual'/'ccorr'.")
293

301    def characterize(self, img_path, peak_path, mask_path, 
302                     imID='im0x', preprocess=True, show=False):
303        """
304        Extract ROIs around peaks, compute features, and select the most 
305        informative subset for downstream classification.
306    
307        Pipeline
308        --------
309        1) Load class masks from ``mask_path`` (``mask1.pkl``..``mask4.pkl``).
310        2) (Optional) Preprocess the image (CLAHE, gamma, normalization).
311        3) Load peaks and prepare data for ROI extraction.
312        4) Extract ROIs (square windows) centered on peaks; ROI size is derived
313           from the mask size.
314        5) Compute features for each ROI (intensity, morphology, correlation,
315           Gaussian fit, etc.).
316        6) Split into train/test and perform forward feature selection (SFS).
317    
318        Parameters
319        ----------
320        img_path : str
321            Path to the source image (e.g., ``.tif``).
322            
323        peak_path : str
324            Path to a pickled peaks file (e.g., a DataFrame with at least 
325            columns ``'X'``, ``'Y'``, and optionally class labels), consumed by
326            ``milab.roi.prep_data``.
327            
328        mask_path : str
329            Directory containing class masks saved as ``maskX.pkl``.
330            The first mask's size is used to derive the ROI half-size.
331            
332        imID : str, optional
333            Image identifier used for labeling/traceability of ROIs. 
334            Default "im0x".
335            
336        preprocess : bool, optional
337            If True, apply CLAHE, gamma correction, and normalization before 
338            ROI extraction. Default True.
339            
340        show : bool, optional
341            If True, display example ROIs and/or intermediate visualizations.
342            Default False.
343    
344        Returns
345        -------
346        None
347            This method updates the object in-place with:
348            - ``self.masks``       : dict of loaded masks
349            - ``self.pimg``        : preprocessed image (if enabled)
350            - ``self.arr, self.df``: prepared data/peaks table
351            - ``self.rois``        : list of (roi_array, imID)
352            - ``self.features``    : feature table (NaNs dropped)
353            - ``self.selection``   : feature selector / chosen feature set
354            - ``self.X_train, self.X_test, self.y_train, self.y_test``
355        """
356        # (1) Load masks
357        self.masks = {}
358        for i in range(1, 5):
359            file_path = os.path.join(mask_path, f"mask{i}.pkl")
360            if not os.path.exists(file_path):
361                raise FileNotFoundError(f"Input file not found: {file_path}")
362            with open(file_path, 'rb') as f:
363                self.masks[i] = pickle.load(f)
364        
365        # Calculate ROI shape based on the mask shape
366        s = int(self.masks[i].shape[0]/2)
367        
368        # (2) Preprocess image
369        if preprocess:
370           self.pimg =  milab.roi.preprocess_image(self.img, 
371                                               apply_clahe=True, 
372                                               gamma=1.2, 
373                                               normalize=True)
374        
375        # (3) Prepare data for ROI extraction
376        self.arr, self.df, peaks = milab.roi.prep_data(
377            img_path, peak_path, min_xy=20, imID=imID, show=False)
378        
379        # keep a reference to the Peaks object
380        self.peaks = peaks
381        
382        # (4) Extract ROIs from image
383        self.rois, self.dfs = [], []
384        rois, df = milab.roi.get_ROIs(im=self.pimg, 
385                                            df=self.df, 
386                                            s=s, 
387                                            norm=False, 
388                                            show=show)
389        
390
391        self.dfs.append(df)        
392        self.rois = [(roi, imID) for roi in rois]        
393        self.dfs = pd.concat(self.dfs, ignore_index=True)
394        
395        # (5) Calculate features
396        self.features, _ = milab.features.get_features(self.rois, self.dfs, 
397                                                       self.masks,
398                                                       show=False)
399
400        self.features = self.features.dropna()
401                
402        # (6) Select features
403        # Split dataset
404        self.X_train, self.X_test, self.y_train, self.y_test = \
405            milab.classifiers.dataset(self.features)
406        
407        # Select features
408        self.selection = milab.classifiers.select_features(self.X_train, 
409                                                           self.y_train, 
410                                                           num=5, 
411                                                           estimator=None)
412        
413        return 
414

416    def correct(image, coords, s=20, method='intensity'):
417        """
418        Refine peak coordinates by re-centering each peak to the local
419        intensity maximum within a square ROI.
420    
421        Parameters
422        ----------
423        image : numpy.ndarray
424            2D grayscale image (HW) in which peaks were detected.
425            
426        coords : pandas.DataFrame
427            Table of peak locations with at least columns ``'X'`` and ``'Y'``.
428            (By convention, ``X`` is the column/index along width, ``Y`` is the
429            row/index along height.)
430            
431        s : int, optional
432            Half-size of the square ROI (in pixels) extracted around each
433            (X, Y) location, i.e. the ROI side is ``2*s+1``. Default ``20``.
434            
435        method : {'intensity'}, optional
436            Refinement strategy. Currently only ``'intensity'`` is supported,
437            which recenters coordinates to the maximum-intensity pixel within
438            the ROI. Default ``'intensity'``.
439    
440        Returns
441        -------
442        pandas.DataFrame
443            A new DataFrame with corrected peak coordinates (and any 
444            accompanying metadata preserved by the underlying ROI routine).
445    
446        Raises
447        ------
448        ValueError
449            If an unsupported ``method`` is requested.
450    
451        Notes
452        -----
453        This function delegates ROI extraction and coordinate refinement to
454        ``milab.roi.get_ROIs`` and returns its updated coordinates table.
455        Make sure your coordinates are within image bounds; peaks too close
456        to the border may yield clipped ROIs depending on ``get_ROIs`` behavior
457        """
458
459        if method == "intensity":
460           _, peaks =  milab.roi.get_ROIs(im=image, df=coords, 
461                                      s=s, norm=False, show=False)
462        else: 
463            raise ValueError("Currently, only the 'intensity'\
464                             method is supported.")
465           
466        return peaks

469    def classify(self, 
470                 data, 
471                 method='gauss_fit', 
472                 target=None, 
473                 estimator=None, 
474                 param_dist=None, 
475                 sfeatures=None):
476        """
477        Classify samples using the selected method.
478    
479        Currently implemented method(s)
480        --------------------------------
481        'rfc' : Random Forest Classifier
482            If no estimator is provided, an RFC is optimized via
483            `milab.classifiers.get_optimal_rfc` using `self.X_train` or
484            `self.y_train` and (optionally) `param_dist`. The optimized model 
485            is then fitted and used to predict on `data`. 
486            If `estimator` is provided (either a fitted classifier instance or 
487            a path to a .pkl file), it is used directly.
488    
489        Parameters
490        ----------
491        data : pandas.DataFrame or numpy.ndarray
492            Feature matrix to classify. If feature selection is used, 
493            the columns should at least include `sfeatures` 
494            (or `self.selection` if omitted).
495            
496        method : str, optional
497            Classification backend. Only 'rfc' is supported at present.
498            (The default 'gauss_fit' is a placeholder and not implemented.)
499            
500        target : array-like, optional
501            Ground-truth labels (used for evaluation inside
502            `milab.classifiers.predicting` if supplied).
503            
504        estimator : sklearn.base.ClassifierMixin or str, optional
505            Either a fitted scikit-learn classifier instance or a filesystem 
506            path to a serialized estimator (.pkl) loadable via `joblib.load`. 
507            If None, a new RFC is optimized and trained.
508            
509        param_dist : dict, optional
510            Hyperparameter search space for optimizing the RFC.
511            Only used when `estimator` is None.
512            
513        sfeatures : list of str, optional
514            Explicit list of feature names to use for training/prediction. 
515            If None, falls back to `self.selection`.
516    
517        Returns
518        -------
519        y_pred : numpy.ndarray
520            Predicted class labels for `data`.
521    
522        """
523        if method == "rfc":
524            # If no estimator is provided, optimize and train a new RFC
525            if estimator is None:
526                # Perform hyperparameter search and get an optimized classifier
527
528                rfc_opt = milab.classifiers.get_optimal_rfc(self.X_train,
529                                                  self.y_train, 
530                                                  param_dist)
531                
532                # Fit the optimized classifier to the training data
533                self.rfc_fit, _ = milab.classifiers.fitting(self.X_train, 
534                                                  self.y_train, 
535                                                  estimator=rfc_opt, 
536                                                  reports=True, 
537                                                  sfeatures=self.selection)
538                
539                # Predict class labels on the input data using the classifier
540                self.y_pred = milab.classifiers.predicting(data, 
541                                                 estimator=self.rfc_fit, 
542                                                 sfeatures=self.selection, 
543                                                 y_test=target)
544            else: 
545                # Load the pre-trained classifier if estimator is a file path
546                if isinstance(estimator, str):
547                    estimator = joblib.load(estimator)
548
549                elif not isinstance(estimator, ClassifierMixin):
550                    # Raise error if estimator is not a valid classifier object
551                    raise ValueError("Provided estimator is invalid.")
552                
553                # Use default selected features if none are provided
554                if sfeatures is None:
555                    sfeatures = self.selection
556                
557                # Predict class labels using the provided classifier
558                self.y_pred = milab.classifiers.predicting(data, 
559                                                 estimator, 
560                                                 sfeatures, 
561                                                 y_test=target)
562                
563        return self.y_pred

302    def characterize(self, img_path, peak_path, mask_path, 
303                     imID='im0x', preprocess=True, show=False):
304        """
305        Extract ROIs around peaks, compute features, and select the most 
306        informative subset for downstream classification.
307    
308        Pipeline
309        --------
310        1) Load class masks from ``mask_path`` (``mask1.pkl``..``mask4.pkl``).
311        2) (Optional) Preprocess the image (CLAHE, gamma, normalization).
312        3) Load peaks and prepare data for ROI extraction.
313        4) Extract ROIs (square windows) centered on peaks; ROI size is derived
314           from the mask size.
315        5) Compute features for each ROI (intensity, morphology, correlation,
316           Gaussian fit, etc.).
317        6) Split into train/test and perform forward feature selection (SFS).
318    
319        Parameters
320        ----------
321        img_path : str
322            Path to the source image (e.g., ``.tif``).
323            
324        peak_path : str
325            Path to a pickled peaks file (e.g., a DataFrame with at least 
326            columns ``'X'``, ``'Y'``, and optionally class labels), consumed by
327            ``milab.roi.prep_data``.
328            
329        mask_path : str
330            Directory containing class masks saved as ``maskX.pkl``.
331            The first mask's size is used to derive the ROI half-size.
332            
333        imID : str, optional
334            Image identifier used for labeling/traceability of ROIs. 
335            Default "im0x".
336            
337        preprocess : bool, optional
338            If True, apply CLAHE, gamma correction, and normalization before 
339            ROI extraction. Default True.
340            
341        show : bool, optional
342            If True, display example ROIs and/or intermediate visualizations.
343            Default False.
344    
345        Returns
346        -------
347        None
348            This method updates the object in-place with:
349            - ``self.masks``       : dict of loaded masks
350            - ``self.pimg``        : preprocessed image (if enabled)
351            - ``self.arr, self.df``: prepared data/peaks table
352            - ``self.rois``        : list of (roi_array, imID)
353            - ``self.features``    : feature table (NaNs dropped)
354            - ``self.selection``   : feature selector / chosen feature set
355            - ``self.X_train, self.X_test, self.y_train, self.y_test``
356        """
357        # (1) Load masks
358        self.masks = {}
359        for i in range(1, 5):
360            file_path = os.path.join(mask_path, f"mask{i}.pkl")
361            if not os.path.exists(file_path):
362                raise FileNotFoundError(f"Input file not found: {file_path}")
363            with open(file_path, 'rb') as f:
364                self.masks[i] = pickle.load(f)
365        
366        # Calculate ROI shape based on the mask shape
367        s = int(self.masks[i].shape[0]/2)
368        
369        # (2) Preprocess image
370        if preprocess:
371           self.pimg =  milab.roi.preprocess_image(self.img, 
372                                               apply_clahe=True, 
373                                               gamma=1.2, 
374                                               normalize=True)
375        
376        # (3) Prepare data for ROI extraction
377        self.arr, self.df, peaks = milab.roi.prep_data(
378            img_path, peak_path, min_xy=20, imID=imID, show=False)
379        
380        # keep a reference to the Peaks object
381        self.peaks = peaks
382        
383        # (4) Extract ROIs from image
384        self.rois, self.dfs = [], []
385        rois, df = milab.roi.get_ROIs(im=self.pimg, 
386                                            df=self.df, 
387                                            s=s, 
388                                            norm=False, 
389                                            show=show)
390        
391
392        self.dfs.append(df)        
393        self.rois = [(roi, imID) for roi in rois]        
394        self.dfs = pd.concat(self.dfs, ignore_index=True)
395        
396        # (5) Calculate features
397        self.features, _ = milab.features.get_features(self.rois, self.dfs, 
398                                                       self.masks,
399                                                       show=False)
400
401        self.features = self.features.dropna()
402                
403        # (6) Select features
404        # Split dataset
405        self.X_train, self.X_test, self.y_train, self.y_test = \
406            milab.classifiers.dataset(self.features)
407        
408        # Select features
409        self.selection = milab.classifiers.select_features(self.X_train, 
410                                                           self.y_train, 
411                                                           num=5, 
412                                                           estimator=None)
413        
414        return

471                 data, 
472                 method='gauss_fit', 
473                 target=None, 
474                 estimator=None, 
475                 param_dist=None, 
476                 sfeatures=None):
477        """
478        Classify samples using the selected method.
479    
480        Currently implemented method(s)
481        --------------------------------
482        'rfc' : Random Forest Classifier
483            If no estimator is provided, an RFC is optimized via
484            `milab.classifiers.get_optimal_rfc` using `self.X_train` or
485            `self.y_train` and (optionally) `param_dist`. The optimized model 
486            is then fitted and used to predict on `data`. 
487            If `estimator` is provided (either a fitted classifier instance or 
488            a path to a .pkl file), it is used directly.
489    
490        Parameters
491        ----------
492        data : pandas.DataFrame or numpy.ndarray
493            Feature matrix to classify. If feature selection is used, 
494            the columns should at least include `sfeatures` 
495            (or `self.selection` if omitted).
496            
497        method : str, optional
498            Classification backend. Only 'rfc' is supported at present.
499            (The default 'gauss_fit' is a placeholder and not implemented.)
500            
501        target : array-like, optional
502            Ground-truth labels (used for evaluation inside
503            `milab.classifiers.predicting` if supplied).
504            
505        estimator : sklearn.base.ClassifierMixin or str, optional
506            Either a fitted scikit-learn classifier instance or a filesystem 
507            path to a serialized estimator (.pkl) loadable via `joblib.load`. 
508            If None, a new RFC is optimized and trained.
509            
510        param_dist : dict, optional
511            Hyperparameter search space for optimizing the RFC.
512            Only used when `estimator` is None.
513            
514        sfeatures : list of str, optional
515            Explicit list of feature names to use for training/prediction. 
516            If None, falls back to `self.selection`.
517    
518        Returns
519        -------
520        y_pred : numpy.ndarray
521            Predicted class labels for `data`.
522    
523        """
524        if method == "rfc":
525            # If no estimator is provided, optimize and train a new RFC
526            if estimator is None:
527                # Perform hyperparameter search and get an optimized classifier
528
529                rfc_opt = milab.classifiers.get_optimal_rfc(self.X_train,
530                                                  self.y_train, 
531                                                  param_dist)
532                
533                # Fit the optimized classifier to the training data
534                self.rfc_fit, _ = milab.classifiers.fitting(self.X_train, 
535                                                  self.y_train, 
536                                                  estimator=rfc_opt, 
537                                                  reports=True, 
538                                                  sfeatures=self.selection)
539                
540                # Predict class labels on the input data using the classifier
541                self.y_pred = milab.classifiers.predicting(data, 
542                                                 estimator=self.rfc_fit, 
543                                                 sfeatures=self.selection, 
544                                                 y_test=target)
545            else: 
546                # Load the pre-trained classifier if estimator is a file path
547                if isinstance(estimator, str):
548                    estimator = joblib.load(estimator)
549
550                elif not isinstance(estimator, ClassifierMixin):
551                    # Raise error if estimator is not a valid classifier object
552                    raise ValueError("Provided estimator is invalid.")
553                
554                # Use default selected features if none are provided
555                if sfeatures is None:
556                    sfeatures = self.selection
557                
558                # Predict class labels using the provided classifier
559                self.y_pred = milab.classifiers.predicting(data, 
560                                                 estimator, 
561                                                 sfeatures, 
562                                                 y_test=target)
563                
564        return self.y_pred
