 132    def normalize(self, signal='amplitude', norm_constant=None, icut=None):
 133        '''
 134        Normalize results of FFT-1D calculation.
 135        
 136        Parameters
 137        ----------
 138        signal : str, optional, default is 'amplitude'
 139            Which signal should be normalized 
 140            (intensity, amplitude, phase, or all).
 141            Intensity and amplitude normalization
 142            usually to 1.0 (default value of norm_constant).
 143            Phase normalization = from (-pi:pi) to (0:2*pi) in order to
 144            eliminate negative values, which cause problems in plotting/saving.
 145        norm_constant : float, optional, default is None
 146            Maximum value/constant, to which the arrays are normalized.
 147            If {None} and what='amplitude' => set to 1.0.
 148            If {None} and what='phase' => set to 2*pi.
 149        icut : float, optional, default is None
 150            Intensity cut.
 151            Example: If {icut} = 200 then all values >200 are set to 200. 
 152
 153        Returns
 154        -------
 155        None
 156            The normalized signal is saved in the properties
 157            of FFT1D object: self.intensity, self.amplitude and/or self.phase.
 158        '''
 159        
 160        # Define local functions ..............................................
 161
 162        def normalize_values(arr, norm_constant):
 163            # Amplitude is a non-negative number
 164            # and so it can be normalized in a standard way.
 165            
 166            # Standard normalization to maximal value
 167            arr = arr/np.max(arr) * norm_constant
 168            arr = arr.astype('float')
 169            return(arr)
 170        
 171        def normalize_angles(arr, norm_constant):
 172            # Phase takes the values in interval (-pi;pi),
 173            # BUT for saving phase as image we need positive values
 174            # THEREFORE, if we want to normalize phase for plotting,
 175            # we have to convert phase to range (0;2*pi) and then normalize.
 176            
 177            # (1) Convert (-pi:pi) to (0:2*pi)
 178            # for the reason explained above
 179            arr = np.where( 
 180                arr < 0, arr + 2*np.pi, arr)
 181            
 182            # (2) Max.phase = upper limit should be ALWAYS 2*pi
 183            # even if this specific number is not in the array.
 184            max_phase = 2*np.pi
 185            
 186            # (3) Standard normalization to maximal value
 187            arr = arr/max_phase * norm_constant
 188            arr = arr.astype('float')
 189            return(arr)
 190        
 191        # Code of the method (after local funcs) ..............................
 192        
 193        # (1) Determine, what to normalize and perform the normalization(s).
 194        if signal == 'all':
 195            # Default: normalize all calculated outputs to default values
 196            self.amplitude = normalize_values(self.amplitude, 1.0)    
 197            self.intensity = normalize_values(self.intensity, 1.0)    
 198            self.phase = normalize_values(self.phase, 2 * np.pi)    
 199        elif signal == 'amplitude':
 200            if norm_constant is None: norm_constant = 1.0
 201            self.amplitude = normalize_values(self.amplitude, norm_constant)
 202        elif signal == 'intensity':
 203            if norm_constant is None: norm_constant = 1.0
 204            self.intensity = normalize_values(self.intensity, norm_constant)
 205        elif signal == 'phase':
 206            if norm_constant is None: norm_constant = 2 * np.pi
 207            normalize_angles(self.phase, norm_constant)
 208        else: 
 209            raise TypeError('Wrong {what} argument!')
 210            
 211        # (3) Perform intensity cut if required (and re-normalize)
 212        if icut is not None:
 213            if signal == 'intensity' or signal == 'all':
 214                arr = self.intensity
 215                arr = np.where(arr > icut, icut, arr)
 216                self.intensity = normalize_values(arr, norm_constant)
 217            elif signal == 'amplitude' or signal == 'all':
 218                arr = self.amplitude
 219                arr = np.where(arr > icut, icut, arr)
 220                self.amplitude = normalize_values(arr, norm_constant)

 327    def find_periodicities(self, 
 328        signal='intensity', peak_height=0.1, decimals=1, verbose=1):
 329        '''
 330        Find dominant frequencies (in reciprocal space)
 331        and corresponding periodicities (in direct space).
 332
 333        Parameters
 334        ----------
 335        signal : str, optional, default is 'amplitude'
 336            Which signal we want to analyze (amplitude or intensity).
 337            Amplitude signal (default) is in line with default imaging.
 338            Intensity signal better emphasizes dominant frequencies.
 339        peak_height : 
 340            Minimal peak height that is considered for calculation.
 341            The {peak_height} argument is employed as {height} argument
 342            of scipy.signal.find_peaks function.
 343        decimals : int, optional, default is 1
 344            Number of decimals for the output frequencies and periodicities.
 345        verbose : int, optional, default is 1
 346            Output verbosity.
 347            If verbose = 0, then nothing is printed on stdout.
 348            If verbose = 1, then print brief lists of dominant
 349            frequencies (peaks of gy) and
 350            periodicities (periodic distances of fx).
 351
 352        Returns
 353        -------
 354        freq, pdist : lists
 355            List of dominant frequencies (in reciprocal space)
 356            and corresponding periodiciies (in direct space).
 357            Moreover the {freq, pdist} values can be printed to stdout
 358            if {verbose} >= 1.
 359        '''
 360        
 361        # Find, which signal we want to analyze ...
 362        if signal == 'intensity':
 363            gy = self.intensity
 364        elif signal == 'amplitude':
 365            gy = self.amplitude
 366        
 367        # Find peaks for given signa ...
 368        peak_indices = scipy.signal.find_peaks(gy, height=peak_height)
 369        peak_positions = self.y[peak_indices[0]]
 370        peak_positions = peak_positions[peak_positions > 0]
 371        
 372        # Store peak_positions = frequencies (f)
 373        # and calculate periodic distance (d)
 374        # by means of BL/reciprocity: f*d = 1
 375        freq = peak_positions
 376        pdist = [round(float(1/f),decimals) for f in freq]
 377        # Print on stdou if requested
 378        if verbose == 1:
 379            print(f'Frequencies   : {freq}')
 380            print(f'Periodicities : {pdist}')
 381        # Return the calculated values
 382        return(freq, pdist)

 872    def calc_radial(arr, center=None):
 873        '''
 874        Calculate radial profile of a 2D array using FFT-consistent coordinates.
 875    
 876        Parameters
 877        ----------
 878        arr : np.ndarray
 879            2D input array.
 880        center : tuple (row, col), optional
 881            Center of the profile in pixels.
 882            If None, uses FFT center (H/2, W/2).
 883    
 884        Returns
 885        -------
 886        R : np.ndarray
 887            Radial distances [pixels].
 888        I : np.ndarray
 889            Mean intensity per radius.
 890        '''
 891        height, width = arr.shape
 892    
 893        # (1) FFT-consistent center
 894        if center is None:
 895            yc = height / 2.0
 896            xc = width / 2.0
 897        else:
 898            yc, xc = center
 899    
 900        # (2) Coordinate grids (CORRECT mapping)
 901        Y, X = np.meshgrid(
 902            np.arange(height) - yc,
 903            np.arange(width) - xc,
 904            indexing='ij')
 905    
 906        # (3) Radial distance map
 907        Rmap = np.sqrt(X**2 + Y**2)
 908    
 909        # (4) Radial bins (1-pixel resolution)
 910        r_max = int(np.floor(Rmap.max()))
 911        R = np.arange(0, r_max, 1)
 912        I = np.zeros_like(R, dtype=float)
 913    
 914        # (5) Mean intensity per radius
 915        for i, r in enumerate(R):
 916            mask = (Rmap >= r) & (Rmap < r + 1)
 917            values = arr[mask]
 918            I[i] = np.mean(values) if values.size else 0.0
 919    
 920        return R, I

1109        '''
1110        Calculate azimuthal profile on a circular ring.
1111    
1112        Parameters
1113        ----------
1114        arr : np.ndarray
1115            2D input array.
1116        radius : float
1117            Radius (in pixels) at which the azimuthal profile is evaluated.
1118        center : tuple (row, col), optional
1119            Center of the pattern. If None, FFT center is used.
1120        bins : int, optional
1121            Number of angular bins (default 360).
1122        dr : float, optional
1123            Half-width of radial detection window in pixels.
1124            Default dr=1.5 gives a 3-pixel thick ring.
1125    
1126        Returns
1127        -------
1128        Theta : np.ndarray
1129            Azimuthal angles [degrees].
1130        I : np.ndarray
1131            Mean intensity per angle bin.
1132        '''
1133        
1134        # (0) Get array height and width
1135        # (Note: the variables MUST have the prefix arr_
1136        # (Reason: {width} is the argument of the function!
1137        arr_height, arr_width = arr.shape
1138    
1139        # (1) FFT-consistent center
1140        if center is None:
1141            yc = arr_height / 2.0
1142            xc = arr_width / 2.0
1143        else:
1144            yc, xc = center
1145    
1146        # (2) Coordinate grids (CORRECT mapping)
1147        Y, X = np.meshgrid(
1148            np.arange(arr_height) - yc,
1149            np.arange(arr_width) - xc,
1150            indexing='ij')
1151    
1152        # (3) Calculate map of distances and map of theta-angles
1153        # (* in fact, we calculate polar coordinates
1154        # (* AI-suggested code with my modifications:
1155        # (  the constant 90 set by trial-and-error to get proper orientation
1156        # (  the modulo % 360 converts everything to positive numbers/angles
1157        Rmap = np.sqrt(X**2 + Y**2)
1158        ThetaMap = (np.degrees(np.arctan2(X, Y)) + 90) % 360
1159    
1160        # (4) Ring mask
1161        ring_mask = (radius-width <= Rmap) & (Rmap <= radius+width)
1162    
1163        # (5) Angular bins
1164        angle_bins = np.linspace(0, 360, bins + 1)
1165        I = np.zeros(bins)
1166        
1167        # (6) Calculate radial profile = intensities along the ring_masks
1168        for i in range(bins):
1169            ang_mask = (
1170                (ThetaMap >= angle_bins[i]) &
1171                (ThetaMap < angle_bins[i + 1]))
1172            mask = ring_mask & ang_mask
1173            values = arr[mask]
1174            I[i] = np.mean(values) if values.size else 0.0
1175        
1176        # (7) Calculate Theta
1177        # (np-trick: average of 1st/2nd, 2nd/3rd ....
1178        Theta = 0.5 * (angle_bins[:-1] + angle_bins[1:])
1179        
1180        # (8) Return the calculated values
1181        return Theta, I
