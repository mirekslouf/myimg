 37class EMDobject:  
 38    '''
 39    Class providing EMDobjects.
 40    
 41    * EMDobject is a HyperSpy object with a few additional methods/properties.
 42    
 43    Technical notes
 44    ---------------
 45    
 46    EMDobject is build by means of three principal components.
 47    
 48    * Initialization:
 49        * __init__ initializes self.hsObject = HyperSpy object
 50        * EMD object = hsObject + self.additional_methods/props
 51    * Re-defined private methods: 
 52        * __getattr__ = delegates access to underlying HyperSpy object methods
 53        * __getitem__ = access to items (in case of list-of-HyperSpy objects)
 54        * __iter__ = access to iteration (in case of list-of-HyperSpy objects)
 55        * __len__ = access to len func (in case of list-of-HyperSpy objects)
 56    * Additional methods/props defined within this class, such as:
 57        * signal, describe, get_scale, save_signal2d ...
 58    '''
 59    
 60    def __init__(self, source_data):
 61        '''
 62        Initialize EMDobject.
 63        '''
 64        
 65        # (1) source_data = str => filename
 66        # => convert to Path and process below in step (2)
 67        if isinstance(source_data, str):
 68            source_data = Path(source_data)
 69        
 70        # (2) source_data = Path object => filename
 71        # => verify if the filename exist, open and save as self.hsObject
 72        if isinstance(source_data, Path):
 73            filename = source_data
 74            if not filename.exists(): 
 75                raise FileNotFoundError(f'File not found: {filename}')
 76            self.hsObject = hs.load(filename, lazy=True)
 77        
 78        # (3) source_data = hsBaseSignals (or perhaps list of hsBaseSignals)
 79        # => save as self.hsObject
 80        elif isinstance(source_data, hsBaseSignal) \
 81            or isinstance(source_data, list):   
 82            self.hsObject = source_data
 83        
 84        # (4) source_data is something else => error
 85        else: 
 86            raise Exception(f'Unknown source data: {source_data}.')
 87
 88
 89    def __getattr__(self, name):
 90        '''
 91        Delegate attribute access to the underlying HyperSpy object.
 92        
 93        Technical note
 94        --------------
 95        __getattr__ method ensures that
 96        unknonwn attributes/methods/properties are passed to self.hsObject
 97        '''
 98        return getattr(self.hsObject, name)
 99    
100    
101    def __getitem__(self, key):
102        '''
103        Delegate item access to the underlying HyperSpy object.
104        
105        Technical note
106        --------------
107        __getattr__ method does not give access to object ITEMS.
108        '''
109        # Get the requested item.
110        result = self.hsObject[key]
111        # Return the item as EMDobject
112        # Reason: EMDobjects (not hsObjects) have access to added methods
113        return EMDobject(result)

116    def __iter__(self):
117        '''
118        Get iteration work - in case of list of HyperSpy objects.
119        '''
120        # We go though the items and return EMDobject for each item
121        # Reason: EMDobjects (not hsObjects) have access to added methods
122        for item in self.hsObject:
123            yield EMDobject(item)

148    def scale(self, out_format=1):
149        '''
150        Return scale from Velox EMD file.
151        
152        * Scale for 2Dsignals = images => pixel size.
153        * Scale for 1Dsignals = spectra => unit on X-axis.
154        * The information about scale is hidden in Velox EMD metadata.
155        '''
156        
157        # (1) Get hsObject, which is saved in self.hsObject
158        hsObject = self.hsObject
159        
160        # (2) If hsObject is a list of hsBaseSignals, take the first signal
161        # (This may happen for multiple singal form BF, DF, HAADF...
162        if isinstance(hsObject, list): hsObject = hsObject[0]
163        
164        # (3) Get pixel size from hsObject metadata
165        number = hsObject.axes_manager.signal_axes[0].scale
166        units = hsObject.axes_manager.signal_axes[0].units
167        
168        # (4) Return the result
169        # (if compact = True, we return the result as single string
170        if out_format == 0:
171            return(number, units)
172        elif out_format == 1:
173            number = f'{number:.5g}'
174            units = units.replace(' ','')
175            number_and_units_as_str = ' '.join((number,units))
176            return(number_and_units_as_str)

180        '''
181        Describe signals in the Velox EMD file, which was read into EMDobject.
182
183        Parameters
184        ----------
185        verbose : int (0,1,2), optional, default is 0
186            The argument determines, how much details we get.

192        '''
193        print('---')
194        extra_space = ('') if verbose == 0 else ('   ')
195        print(f'Signal {extra_space}: {EMDmetadata.signal_description(self)}')
196        if verbose >= 1:
197            print(f'Detectors : {EMDmetadata.list_of_detectors(self)}')
198        if verbose >= 2:

204        '''
205        Show 2D-signal from the Velox EMD file.

207        # (1) Get signal

223        '''
224        Save 2D-signal from the Velox EMD file.
225        '''
226        # (1) Get signal
227        if item is None:

229        else:
230            sig = self[item].data

232        arr = sig.astype(np.uint16).compute()

234        img = PIL.Image.fromarray(arr, mode='I;16')

238    def close(self):
239        '''
240        Close EMDobject = release its links and prepare for destruction.
241        
242        Returns
243        -------
244        None
245            The object is now empty.
246            Its complete destruction must be done in the main code, 
247            by calling `del(object)`.
248        
249        Technical notes
250        ---------------
251        * General rule: In Python, the objects don't kill themselves -
252          they just drop what they own.
253        * Therefore, this is just a *preparation* for object destruction,
254          which must be done manually, in main program,
255          using del(object).
256        * Nevertheless, the object destruction may be useful
257          in Spyder or Jupyter, where the opened EMD objects occupy memory
258          and/or they may lock EMD files during repeated runs of a script.
259        '''
260        self.hsObject = None

286    @classmethod 
287    def rename_orig(cls, vdir, idigits=3, validate=False):
288        # If vdir is string, convert it to Path object.
289        if isinstance(vdir, str): vdir = Path(vdir)
290        # Define EMD files withing {vdir} directory.
291        # (only Velox EMD files of type: 0001 - something.emd
292        emd_files = vdir.rglob('???? - ?*.emd')
293        # Go through EMD files, and rename them.
294        for file in sorted(emd_files):
295            new_name = cls.short_name(file, fullpath=True)
296            if validate is True:
297                print(f'{file} => {new_name}')
298            else:
299                shutil.move(file, new_name)

303    def rename_exported(cls, vdir, idigits=3, validate=False):
304        # If vdir is string, convert it to Path object.
305        if isinstance(vdir, str): vdir = Path(vdir)
306        # Define image files within {vdir} directory.
307        # (only Velox exported files of type: 0001 - something.png|tif|txt
308        img_files = itertools.chain(
309            vdir.rglob('???? - ?*.png'),
310            vdir.rglob('???? - ?*.tif'),
311            vdir.rglob('???? - ?*.txt'))
312        # Go through EMD files, and rename them.
313        for file in sorted(img_files):
314            new_name = cls.short_name( 
315                file, fullpath=True, last_part_of_name=True)
316            if validate is True:
317                print(f'{file} => {new_name}')
318            else:
319                shutil.move(file, new_name)

322    @classmethod
323    def short_name(cls, file, 
324                   idigits=3, last_part_of_name=False,
325                   fullpath=False, subdirs=None, extension=False):
326        # Convert file argument to Path object - for the sake of consistency
327        if isinstance(file, str): file = Path(file)
328        # Get stem of the filename
329        # (this will be gradually changed to new_name
330        orig_name = file.stem
331        # Convert the initial separator between file number and the rest
332        new_name = orig_name.replace(' - ','_')
333        # Remove space between magnification and 'x'/'kx'
334        new_name = new_name.replace(' x','x')
335        new_name = new_name.replace(' kx','kx')
336        # Remove space between camera-lenght and 'mm'
337        new_name = new_name.replace(' mm','mm')
338        # Correct possible error when the JobID is inserted as H66_
339        new_name = new_name.replace('_ ','_')
340        # Convert all other whitespace to underscore
341        new_name = new_name.replace(' ','_')
342        # Split to components
343        all_elements = new_name.split('_')
344        # New name contains only the first three elements
345        name_elements = all_elements[0:3]
346        # Add the last element (detector name) if requested
347        if last_part_of_name:
348            # We sould add the last element only if it exists!
349            # (the name can have just 3 elements: {number} {jobID} {mag/CC}
350            if len(all_elements) > 3:
351                name_elements.append(all_elements[-1])
352        # Redefine the number of digits in the first element
353        name_elements[0] = f'{int(name_elements[0]):0{idigits}d}'
354        # Combine all requested element
355        new_name = '_'.join(name_elements)
356        # Add subdirs and/or extension to the new_name if requested
357        if fullpath is True:
358            # Example: if fullpath = True => new_name = c:/d1/d2/new_name.emd
359            new_name = Path(file.parent, new_name)        # add parent dir
360            new_name = new_name.with_suffix(file.suffix)  # add extension
361        elif (subdirs is not None) and (fullpath is False):
362            # Assumption: file is c:/d1/d2/d3/orig_name.emd
363            # Example: if subdirs = 2 => new_name = d2/d3/new_name
364            new_name = cls.parent_dirs(file, subdirs) / new_name
365        # Add extension of requeste
366        if (extension is True) and (fullpath is False):
367            # Example: if extension == True => new_name = new_name.emd
368            # (NOT done if fullpath was requested =>  extension already added
369            new_name = Path(new_name).with_suffix(file.suffix)
370        # Return the shortened name
371        return(new_name)
372

374    @classmethod 
375    def parent_dirs(path, n):
376        # Get parent dir and split it to parts.
377        # (input: c:\d1\d2\d3\file.txt => c:, d1, d2, d3
378        parents = path.parent.parts
379        # Get list of last n parent dirs, join them to path
380        # (input {c:, d1, d2, d3} + {2} => d2\d3
381        last_n_parents = Path(*parents[-n:])
382        # Return the path of n last parent dirs
383        return(last_n_parents)

386    @classmethod
387    def describe(cls, vdir, fullpath=False, 
388                 output_print=True, output_string=False, output_file=None):
389        # If vdir is string, convert it to Path object.
390        if isinstance(vdir, str): vdir = Path(vdir)
391        # Prepare list/generator for emd_files
392        emd_files = vdir.rglob('*.emd')
393        # Prepare lists for the parameters we want to collect
394        filenames = []
395        signals   = []
396        apertures = []
397        detectors = []
398        # Go through EMD files,
399        # collect selected parameters and save them to lists.
400        for file in emd_files:
401            if fullpath is True:
402                filenames.append(file)
403            else:
404                filenames.append(file.relative_to(vdir))
405            hsObject = hs.load(file, lazy=True)
406            signals.append( EMDmetadata.signal_description(hsObject) )
407            apertures.append( EMDmetadata.list_of_apertures(hsObject) )
408            detectors.append( EMDmetadata.list_of_detectors(hsObject) )
409            # Release hsObject from memory
410            del(hsObject)
411        # Release hsObjects from memory
412        gc.collect()
413        # Convert the saved lists into a dataframe.
414        df = pd.DataFrame({
415            'Filename': filenames, 'Signal': signals,  
416            'Apertures[um]': apertures, 'Detector(s)': detectors})
417        # Convert the dataframe to a (nicely formatted) table.
418        table = tabulate.tabulate(df,showindex=False, headers=df.columns)
419        # Show/save final result
420        # (1) If output_print=True (default) => print table to stdout.
421        if output_print is True:
422            print(table)
423        # (2) If ouput_string=True (optional) => return table as string.
424        if output_string is True:
425            return(table)
426        # (3) If output_file is specified (optional) => save table to file.
427        if output_file is not None:
428            with open(output_file, 'w') as fh: fh.writelines(table)

469    @classmethod
470    def signal_description(cls, hsObject):
471        
472        # (1) {hsObject} argument = datafile with one hsBaseSignal
473        #     => describe the signal
474        if EMDmetadata.is_single_signal(hsObject):
475            # Determine detector_type (TEM or STEM or EDS)
476            if cls.is_TEM(hsObject):
477                detector_type = 'TEM'
478            elif cls.is_STEM(hsObject):
479                detector_type = 'STEM'
480            elif cls.is_EDS_spectrum(hsObject):
481                detector_type = 'EDS'
482                return('EDS:spectrum')
483            else:
484                detector_type = 'Unknown'
485            # Determine signal_type (image or diffractogram)
486            if cls.is_Image(hsObject):
487                signal_type = 'image'
488            elif cls.is_Diffraction(hsObject):
489                signal_type = 'diff'
490            else:
491                signal_type = 'Unknown'           
492            return(f'{detector_type}:{signal_type}')
493        
494        # (2) {hsObject} argument = datafile with list of hsBaseSignals
495        #     => describe just the first signal
496        #     => assupmtion: all signals are analogous (typically BF,DF,HAADF)
497        elif EMDmetadata.is_list_of_signals(hsObject):
498            detector_type_and_signal = cls.signal_description(hsObject[0])
499            return(detector_type_and_signal)
500        
501        # (3) {hsObject} is something else
502        #     => return a brief message that this type of signal is unknown
503        else: return "Uknown signal."

506    @classmethod
507    def list_of_detectors(cls, hsObject, concise=True):
508        
509        # (1) {hsObject} argument = datafile with one hsBaseSignal
510        if EMDmetadata.is_single_signal(hsObject):
511            sig  = hsObject.metadata.General.title
512            sdim = hsObject.axes_manager.signal_dimension
513            if concise:
514                if sig in cls.shorter_detector_names.keys():
515                    sig = cls.shorter_detector_names[sig]
516                return(sig)
517            else:
518                return(f'{sdim}D:{sig}')
519        
520        # (2) {hsObject} argument = datafile with list of hsBaseSignals
521        #     => describe the individual signals one by one 
522        elif EMDmetadata.is_list_of_signals(hsObject):
523            list_of_signals = [
524                cls.list_of_detectors(sig, concise=concise)
525                for sig in hsObject ]
526            joined_list_of_signals = '[' + ' '.join(list_of_signals) + ']'
527            return joined_list_of_signals
528        
529        # (3) {hsObject} is something else
530        #     => return a brief message that this type of signal is unknown
531        else: return "Unknown signal."

534    @classmethod
535    def list_of_apertures(cls, hsObject):
536        
537        # (1) {hsObject} argument = datafile with one hsBaseSignal
538        if EMDmetadata.is_single_signal(hsObject):
539            # Read apertures
540            try:
541                apertures = hsObject.original_metadata.Optics.Apertures
542            except AttributeError:
543                return "No apertures in original_metadata."
544            # Go through all apertures + add them to list
545            ap_list = []
546            for ap_key in sorted(apertures.keys()):
547                ap = apertures[ap_key]
548                if ap.Type != 'None':
549                    ap_diameter = round(float(ap.Diameter)*1e6)
550                    ap_list.append(f'{ap.Name}:{ap_diameter:>3d}')
551                else:
552                    ap_list.append(f'{ap.Name}:---')
553            # Convert the list to a final string with all apertures used
554            ap_list = '[' + ' '.join(ap_list) + ']'
555            return(ap_list)
556        
557        # (2) {hsObject} argument = datafile with list of hsBaseSignals
558        elif EMDmetadata.is_list_of_signals(hsObject):
559            ap_list = cls.list_of_apertures(hsObject[0])
560            return(ap_list)
561        
562        # (3) {hsObject} is something else
563        else: return('Unknown signal.')

 40    Class providing EMDobjects.
 41    
 42    * EMDobject is a HyperSpy object with a few additional methods/properties.
 43    
 44    Technical notes
 45    ---------------
 46    
 47    EMDobject is build by means of three principal components.
 48

 51        * EMD object = hsObject + self.additional_methods/props
 52    * Re-defined private methods: 
 53        * __getattr__ = delegates access to underlying HyperSpy object methods
 54        * __getitem__ = access to items (in case of list-of-HyperSpy objects)

323    @classmethod
324    def short_name(cls, file, 
325                   idigits=3, last_part_of_name=False,
326                   fullpath=False, subdirs=None, extension=False):
327        # Convert file argument to Path object - for the sake of consistency
328        if isinstance(file, str): file = Path(file)
329        # Get stem of the filename
330        # (this will be gradually changed to new_name
331        orig_name = file.stem
332        # Convert the initial separator between file number and the rest
333        new_name = orig_name.replace(' - ','_')
334        # Remove space between magnification and 'x'/'kx'
335        new_name = new_name.replace(' x','x')
336        new_name = new_name.replace(' kx','kx')
337        # Remove space between camera-lenght and 'mm'
338        new_name = new_name.replace(' mm','mm')
339        # Correct possible error when the JobID is inserted as H66_
340        new_name = new_name.replace('_ ','_')
341        # Convert all other whitespace to underscore
342        new_name = new_name.replace(' ','_')
343        # Split to components
344        all_elements = new_name.split('_')
345        # New name contains only the first three elements
346        name_elements = all_elements[0:3]
347        # Add the last element (detector name) if requested
348        if last_part_of_name:
349            # We sould add the last element only if it exists!
350            # (the name can have just 3 elements: {number} {jobID} {mag/CC}
351            if len(all_elements) > 3:
352                name_elements.append(all_elements[-1])
353        # Redefine the number of digits in the first element
354        name_elements[0] = f'{int(name_elements[0]):0{idigits}d}'
355        # Combine all requested element
356        new_name = '_'.join(name_elements)
357        # Add subdirs and/or extension to the new_name if requested
358        if fullpath is True:
359            # Example: if fullpath = True => new_name = c:/d1/d2/new_name.emd
360            new_name = Path(file.parent, new_name)        # add parent dir
361            new_name = new_name.with_suffix(file.suffix)  # add extension
362        elif (subdirs is not None) and (fullpath is False):
363            # Assumption: file is c:/d1/d2/d3/orig_name.emd
364            # Example: if subdirs = 2 => new_name = d2/d3/new_name
365            new_name = cls.parent_dirs(file, subdirs) / new_name
366        # Add extension of requeste
367        if (extension is True) and (fullpath is False):
368            # Example: if extension == True => new_name = new_name.emd
369            # (NOT done if fullpath was requested =>  extension already added
370            new_name = Path(new_name).with_suffix(file.suffix)
371        # Return the shortened name
372        return(new_name)

515                if sig in cls.shorter_detector_names.keys():
516                    sig = cls.shorter_detector_names[sig]
517                return(sig)
518            else:
519                return(f'{sdim}D:{sig}')
520        
521        # (2) {hsObject} argument = datafile with list of hsBaseSignals
522        #     => describe the individual signals one by one 
523        elif EMDmetadata.is_list_of_signals(hsObject):
524            list_of_signals = [
525                cls.list_of_detectors(sig, concise=concise)
526                for sig in hsObject ]
527            joined_list_of_signals = '[' + ' '.join(list_of_signals) + ']'
528            return joined_list_of_signals
529        
530        # (3) {hsObject} is something else
531        #     => return a brief message that this type of signal is unknown
532        else: return "Unknown signal."

535    @classmethod
536    def list_of_apertures(cls, hsObject):
537        
538        # (1) {hsObject} argument = datafile with one hsBaseSignal
539        if EMDmetadata.is_single_signal(hsObject):
540            # Read apertures
541            try:
542                apertures = hsObject.original_metadata.Optics.Apertures
543            except AttributeError:
544                return "No apertures in original_metadata."
545            # Go through all apertures + add them to list
546            ap_list = []
547            for ap_key in sorted(apertures.keys()):
548                ap = apertures[ap_key]
549                if ap.Type != 'None':
550                    ap_diameter = round(float(ap.Diameter)*1e6)
551                    ap_list.append(f'{ap.Name}:{ap_diameter:>3d}')
552                else:
553                    ap_list.append(f'{ap.Name}:---')
554            # Convert the list to a final string with all apertures used
555            ap_list = '[' + ' '.join(ap_list) + ']'
556            return(ap_list)
557        
558        # (2) {hsObject} argument = datafile with list of hsBaseSignals
559        elif EMDmetadata.is_list_of_signals(hsObject):
560            ap_list = cls.list_of_apertures(hsObject[0])
561            return(ap_list)
562        
563        # (3) {hsObject} is something else
564        else: return('Unknown signal.')
