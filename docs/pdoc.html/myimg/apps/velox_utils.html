 37class EMDobject:  
 38    '''
 39    Class providing EMDobjects.
 40    
 41    * EMDobject provides access to the content of Velox EMD files.
 42    * EMDobject is a HyperSpy object with a few additional methods/properties.
 43    * EMDobject can use all HyperSpy methods + our own methods defined below.
 44    
 45    Example - use EMDobject to save an image from EMD file:
 46    
 47    >>> # Standard import of MyImg + package to read Velox files 
 48    >>> import myimg.api as mi
 49    >>> Velox = mi.Apps.import_Velox_utils
 50    >>>
 51    >>> # Read an EMD file and print what is inside
 52    >>> emd = Velox.EMDobject(r'some.emd')
 53    >>> emd.describe(verbose=3)
 54    >>>
 55    >>> # Show and save an image from EMD file
 56    >>> emd.signal2d_show(cmap='magma')
 57    >>> emd.signal2d_save('some.png')
 58    '''
 59
 60    # Technical notes
 61    # ---------------    
 62    # EMDobject is build by means of three principal components.
 63    # 1. Initialization:
 64    #     * __init__ initializes self.hsObject = HyperSpy object
 65    #     * EMD object = hsObject + self.additional_methods/props
 66    # 2. Re-defined private methods: 
 67    #     * __getattr__ = delegates access to underlying HyperSpy object methods
 68    #     * __getitem__ = access to items (in case of list-of-HyperSpy objects)
 69    #     * __iter__ = access to iteration (in case of list-of-HyperSpy objects)
 70    #     * __len__ = access to len func (in case of list-of-HyperSpy objects)
 71    # 3. Additional methods/props defined within this class, such as:
 72    #     * signal, describe, get_scale, save_signal2d ...
 73    
 74    def __init__(self, source_data):
 75        '''
 76        Initialize {EMDobject} = a {HyperSpy} object with additional methods.
 77
 78        Parameters
 79        ----------
 80        source_data : str or PathLike or HyperSpy object
 81            Source data for the {EMDobject}.
 82            Typically, {source_data} is a filename of existing Velox EMD file.
 83            Alternatively, {source_data} can be a HyperSpy object,
 84            which was used to read the Velox EMD file.
 85
 86        Returns
 87        -------
 88        EMDobject
 89        '''
 90        
 91        # (1) source_data = str => filename
 92        # => convert to Path and process below in step (2)
 93        if isinstance(source_data, str):
 94            source_data = Path(source_data)
 95        
 96        # (2) source_data = Path object => filename
 97        # => verify if the filename exist, open and save as self.hsObject
 98        if isinstance(source_data, Path):
 99            filename = source_data
100            if not filename.exists(): 
101                raise FileNotFoundError(f'File not found: {filename}')
102            self.hsObject = hs.load(filename, lazy=True)
103        
104        # (3) source_data = hsBaseSignals (or perhaps list of hsBaseSignals)
105        # => save as self.hsObject
106        elif isinstance(source_data, hsBaseSignal) \
107            or isinstance(source_data, list):   
108            self.hsObject = source_data
109        
110        # (4) source_data is something else => error
111        else: 
112            raise Exception(f'Unknown source data: {source_data}.')

203    def scale(self, out_format=1):
204        '''
205        Scale from Velox EMD file.
206        
207        * Scale for 2Dsignals = images => pixel size.
208        * Scale for 1Dsignals = spectra => unit on X-axis.
209        * The information about scale is hidden in Velox EMD metadata.
210        
211        Parameters
212        ----------
213        out_format : int, optional, default is 1
214            If {out_format} = 1, we get output like '1.25 nm'.
215            If {out_format} = 0, we get output like (1.25, 'nm').
216            
217        Returns
218        -------
219        str or tuple
220            Scale of the signal (such as pixel size, unit on axis).
221            The form of the output depend on {out_format} argument.
222            The various types of scales and output formats are described above.
223        '''
224        
225        # (1) Get hsObject, which is saved in self.hsObject
226        hsObject = self.hsObject
227        
228        # (2) If hsObject is a list of hsBaseSignals, take the first signal
229        # (This may happen for multiple singal form BF, DF, HAADF...
230        if isinstance(hsObject, list): hsObject = hsObject[0]
231        
232        # (3) Get pixel size from hsObject metadata
233        number = hsObject.axes_manager.signal_axes[0].scale
234        units = hsObject.axes_manager.signal_axes[0].units
235        
236        # (4) Return the result
237        # (if compact = True, we return the result as single string
238        if out_format == 0:
239            return(number, units)
240        elif out_format == 1:
241            number = f'{number:.5g}'
242            units = units.replace(' ','')
243            number_and_units_as_str = ' '.join((number,units))
244            return(number_and_units_as_str)

272        '''
273        Show 2D-signal from the Velox EMD file.
274        
275        Parameters
276        ----------
277        out_file : str or path, optional, default is None
278            If {out_file} is not None, the plot of the signal will be saved
279            to the image file with name {out_file} and resolution {out_dpi}.

295        if item is None: 
296            sig = self.data
297        else:
298            sig = self[item].data
299        # (2) Plot signal
300        plt.imshow(sig, **kwargs)
301        plt.tight_layout()
302        # (3) Save the plot if requested
303        if out_file is not None:
304            plt.savefig(out_file, dpi=out_dpi)

310        '''
311        Save 2D-signal from the Velox EMD file as a 16-bit image file.
312        
313        Parameters
314        ----------
315        out_file : str or path
316            Name of the output file.
317            The extension determines the type of the file.
318        item : None or int, optional, default is None
319            Obligatory if the original EMD file contains series of images.
320            Example: EMD file contains three images [BF DF HAADF]
321            and we set {item} = 2; in this case, the function will save
322            the HAADF (because in our examlple: item=0 is BF, item=1 is DF,
323            and item=2 is HAADF).
324        '''
325        # (1) Get signal
326        if item is None: 
327            sig = self.data
328        else:
329            sig = self[item].data
330        # (2) Convert signal to numpy.array
331        arr = sig.astype(np.uint16).compute()
332        # (3) Convert array to image + save to {out_file}
333        img = PIL.Image.fromarray(arr, mode='I;16')
334        img.save(out_file)

473    @classmethod
474    def short_name(cls, file, 
475                   idigits=3, last_part_of_name=False,
476                   fullpath=False, subdirs=None, extension=False):
477        # Convert file argument to Path object - for the sake of consistency
478        if isinstance(file, str): file = Path(file)
479        # Get stem of the filename
480        # (this will be gradually changed to new_name
481        orig_name = file.stem
482        # Start with converting the all chars in filename to lowercase
483        # (this standardization can save a lot of troubles
484        new_name = orig_name.lower()
485        # Convert the initial separator between file number and the rest
486        new_name =new_name.replace(' - ','_')
487        # Remove space between magnification and 'x'/'kx'
488        new_name = new_name.replace(' x','x')
489        new_name = new_name.replace(' kx','kx')
490        # Remove space between camera-lenght and 'mm'
491        new_name = new_name.replace(' mm','mm')
492        # Correct possible error when the JobID is inserted as H66_
493        new_name = new_name.replace('_ ','_')
494        # Change special names of EDS results exported to CSV
495        # (* These files can be created in Velox by:
496        # (  Velox-Processing - Menu - EDS - Export quantification details
497        # (* We hard-ren:ame these special files in the following way
498        # (  exchange initial '-' for ' ' => to separate {last_part_of_name}
499        # (  do not use '_' in the text   => not to split {last_part_of_name}
500        # (  => then {last_part_of_name} is compatible with further processing 
501        new_name = new_name.replace('-eds spectrum-composition', ' eds1-elem')
502        new_name = new_name.replace('-eds spectrum-lines',       ' eds2-lines')
503        # Convert all other whitespace to underscore
504        new_name = new_name.replace(' ','_')
505        # Split to components
506        all_elements = new_name.split('_')
507        # New name contains only the first three elements
508        name_elements = all_elements[0:3]
509        # Add the last element (detector name) if requested
510        if last_part_of_name:
511            # We sould add the last element only if it exists!
512            # (the name can have just 3 elements: {number} {jobID} {mag/CC}
513            if len(all_elements) > 3:
514                name_elements.append(all_elements[-1])
515        # Redefine the number of digits in the first element
516        name_elements[0] = f'{int(name_elements[0]):0{idigits}d}'
517        # Combine all requested element
518        new_name = '_'.join(name_elements)
519        # Add subdirs and/or extension to the new_name if requested
520        if fullpath is True:
521            # Example: if fullpath = True => new_name = c:/d1/d2/new_name.emd
522            new_name = Path(file.parent, new_name)        # add parent dir
523            new_name = new_name.with_suffix(file.suffix)  # add extension
524        elif (subdirs is not None) and (fullpath is False):
525            # Assumption: file is c:/d1/d2/d3/orig_name.emd
526            # Example: if subdirs = 2 => new_name = d2/d3/new_name
527            new_name = cls.parent_dirs(file, subdirs) / new_name
528        # Add extension of requeste
529        if (extension is True) and (fullpath is False):
530            # Example: if extension == True => new_name = new_name.emd
531            # (NOT done if fullpath was requested =>  extension already added
532            new_name = Path(new_name).with_suffix(file.suffix)
533        # Return the shortened name
534        return(new_name)

537    @classmethod 
538    def parent_dirs(path, n):
539        # Get parent dir and split it to parts.
540        # (input: c:\d1\d2\d3\file.txt => c:, d1, d2, d3
541        parents = path.parent.parts
542        # Get list of last n parent dirs, join them to path
543        # (input {c:, d1, d2, d3} + {2} => d2\d3
544        last_n_parents = Path(*parents[-n:])
545        # Return the path of n last parent dirs
546        return(last_n_parents)

549    @classmethod
550    def describe(cls, vdir, fullpath=False, 
551                 output_print=True, output_string=False, output_file=None):
552        # If vdir is string, convert it to Path object.
553        if isinstance(vdir, str): vdir = Path(vdir)
554        # Prepare list/generator for emd_files
555        emd_files = vdir.rglob('*.emd')
556        # Prepare lists for the parameters we want to collect
557        filenames = []
558        signals   = []
559        apertures = []
560        detectors = []
561        # Go through EMD files,
562        # collect selected parameters and save them to lists.
563        for file in emd_files:
564            if fullpath is True:
565                filenames.append(file)
566            else:
567                filenames.append(file.relative_to(vdir))
568            hsObject = hs.load(file, lazy=True)
569            signals.append( EMDmetadata.signal_description(hsObject) )
570            apertures.append( EMDmetadata.list_of_apertures(hsObject) )
571            detectors.append( EMDmetadata.list_of_detectors(hsObject) )
572            # Release hsObject from memory
573            del(hsObject)
574        # Release hsObjects from memory
575        gc.collect()
576        # Convert the saved lists into a dataframe.
577        df = pd.DataFrame({
578            'Filename': filenames, 'Signal': signals,  
579            'Apertures[um]': apertures, 'Detector(s)': detectors})
580        # Convert the dataframe to a (nicely formatted) table.
581        table = tabulate.tabulate(df,showindex=False, headers=df.columns)
582        # Show/save final result
583        # (1) If output_print=True (default) => print table to stdout.
584        if output_print is True:
585            print(table)
586        # (2) If ouput_string=True (optional) => return table as string.
587        if output_string is True:
588            return(table)
589        # (3) If output_file is specified (optional) => save table to file.
590        if output_file is not None:
591            with open(output_file, 'w') as fh: fh.writelines(table)

701        # (1) {hsObject} argument = datafile with one hsBaseSignal
702        if EMDmetadata.is_single_signal(hsObject):
703            # Read apertures

705                apertures = hsObject.original_metadata.Optics.Apertures
706            except AttributeError:
707                return "No apertures in original_metadata."
708            # Go through all apertures + add them to list
709            ap_list = []
710            for ap_key in sorted(apertures.keys()):
711                ap = apertures[ap_key]
712                if ap.Type != 'None':
713                    ap_diameter = round(float(ap.Diameter)*1e6)
714                    ap_list.append(f'{ap.Name}:{ap_diameter:>3d}')
715                else:
716                    ap_list.append(f'{ap.Name}:---')
717            # Convert the list to a final string with all apertures used
718            ap_list = '[' + ' '.join(ap_list) + ']'
719            return(ap_list)
720        
721        # (2) {hsObject} argument = datafile with list of hsBaseSignals
722        elif EMDmetadata.is_list_of_signals(hsObject):
723            ap_list = cls.list_of_apertures(hsObject[0])
724            return(ap_list)

726        # (3) {hsObject} is something else
727        else: return('Unknown signal.')
