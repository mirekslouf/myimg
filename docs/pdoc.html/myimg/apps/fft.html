 28    def __init__(self, img):
 29        '''
 30        Initialize FFT object.
 31        
 32        >>> # Simple usage of FFT objects
 33        >>> import myimg.api as mi        # standard import of myimg
 34        >>> img = mi.MyImage('some.png')  # open an image using myimg.api
 35        >>> fft = my.Apps.FFT(img)        # calculate FFT of the img object
 36        >>> fft.show(cmap='magma')        # show the result
 37
 38        Parameters
 39        ----------
 40        img : image (array or str or path-like or MyImage object)
 41            The 2D object, from which we will calculate FFT
 42            = 2D-DFFT = 2D discrete fast Fourier transform.
 43
 44        Returns
 45        -------
 46        FFT object.
 47        
 48        Technical details
 49        -----------------
 50        * FFT object, 3 basic attributes: FFT.fft (array of complex numbers),
 51          FFT.intensity (array of intensities = magnitudes = real numbers)
 52          and FFT.phase (array of phases = angles in range -pi:pi).
 53        * FFT object is pre-processed in the sense that the intensity center
 54          is shifted to the center of the array (using scipy.fftpack.fftshift).
 55        * FFT object carries the information about calibration (pixel-size),
 56          on condition it was created from MyImage object (the typical case).
 57        '''
 58        
 59        # (1) Process the first argument = image
 60        # (the first argument = input array
 61        # (it can be: np.array, MyImage object, PIL object or path-to-image
 62        if type(img) == np.ndarray:
 63            # img comes as numpy array
 64            # (the simplest case - just assign img to arr
 65            arr = img
 66        elif type(img) == myimg.api.MyImage:
 67            # img comes as MyImage object
 68            # (check the image type and convert it to array if possible
 69            if img.itype in ('binary','gray','gray16'):
 70                arr = np.asarray(img.img)
 71            else:
 72                print('Fourier transform only for binary or grayscale images!')
 73                sys.exit()
 74        elif type(img) == str:
 75            # image comes as str => we assume it is an image name
 76            # TODO: check the image type
 77            img = Image.open(img)
 78            arr = np.asarray(img)
 79        else:
 80            print('Unknown image type!')
 81            sys.exit()
 82        
 83        # (2) Calculate FFT, shift the origin, and save the result
 84        arr = scipy.fftpack.fft2(arr)
 85        arr = scipy.fftpack.fftshift(arr)
 86        self.fft = arr
 87        self.intensity = np.abs(arr)
 88        self.phase = np.angle(arr)
 89        
 90        # (3) Save the information about pixel size, if available
 91        if type(img) == myimg.api.MyImage:
 92            # TODO: extend rec.units, add units here
 93            # TODO: consider adding set_scale, scalebar to fft package
 94            # self.recpixsize = 1 / (img.width * img.pixsize.number)
 95            pass
 96        else:
 97            self.recpixsize = None

100    def normalize(self, what='intensity', itype='16bit', icut=None):
101        '''
102        Normalize results of fft calculation.
103        
104        Parameters
105        ----------
106        what : str, optional, default is 'intensity'
107            What result should be normalized - 'intensity' or 'phase'.
108            Intensity normalization = from arbitrary scale to given itype.
109            Phase normalization = from (-pi:pi) to (0:2*pi) in order to
110            eliminate negative values, which cause problems in plotting/saving.
111        itype : str, optional, default is '16bit'
112            Format of the normalized fft arrays (that can be saved as images).
113            If '16bit' (default), the images will be 16-bit.
114            If '8bit' (less suitable, narrow range) the images will be 8-bit.
115        icut : TYPE, optional
116            DESCRIPTION. The default is None.
117
118        Returns
119        -------
120        None.
121        '''
122        
123        # Define local functions
124
125        def normalize_intensity(norm_constant):
126            # Intensity is a non-negative number
127            # and so it can be normalized in a standard way.
128            
129            # (1) Dermine max.value = maximal intenstity in the array.
130            max_intensity = np.max(self.intensity)
131            
132            # (2) Standard normalization to maximal value
133            # BUT taking into account rounding and type of the final array.
134            # (a) Standard normalization
135            self.intensity = self.intensity/max_intensity * norm_constant
136            # (b) Adjusting type of the normalized array
137            # If the normalization constant is an integer value,
138            # ..round the result to int and convert the array to integers
139            # ..this is important for the smooth converting of arrays to images
140            if type(norm_constant) == int:
141                self.intensity = np.round(self.intensity).astype('int')
142        
143        def normalize_phase(norm_constant):
144            # Phase takes the values in interval (-pi;pi),
145            # BUT for saving phase as image we need positive values
146            # THEREFORE, if we want to normalize phase for plotting,
147            # we have to convert phase to range (0;2*pi) and then normalize.
148            
149            # (1) Convert (-pi:pi) to (0:2*pi)
150            # for the reason explained above
151            self.phase = np.where( 
152                self.phase < 0, self.phase + 2*np.pi, self.phase)
153            
154            # (2) Max.phase = upper limit should be ALWAYS 2*pi
155            # even if this specific number is not in the array.
156            max_phase = 2*np.pi
157            
158            # (3) Standard normalization to maximal value
159            # BUT taking into account rounding and type of the final array.
160            # (a) Standard normalization
161            self.phase = self.phase/max_phase * norm_constant
162            # (b) Adjusting type of the normalized array
163            # If the normalization constant is an integer value,
164            # ..round the result to int and convert the array to integers
165            # ..this is important for the smooth converting of arrays to images
166            if type(norm_constant) == 'int':
167                self.phase = np.round(self.phase).astype('int')
168        
169        # Code of the method itself (after defning local functions)
170        
171        # (1) Calculate normalization constant        
172        if itype == '16bit': norm_constant = 2**16 - 1
173        elif itype == '8bit': norm_constant = 2**8 - 1 
174        else: norm_constant = itype
175        
176        # (2) Determine, what to normalize and perform the normalization(s).
177        if what == 'intensity':
178            normalize_intensity(norm_constant)
179        elif what == 'phase':
180            normalize_phase(norm_constant)
181        else:  # Exit if what argument was wrong. 
182            print('myimg.apps.fft.convert_to_16big -> wrong what argument!')
183            sys.exit()
184            
185        # (3) Perform intensity cut if required (and re-normalize)
186        if (what == 'intensity') and (icut is not None):
187            arr = self.intensity
188            self.intensity = np.where(arr > icut, icut, arr)
189            normalize_intensity(norm_constant)

192    def show(self, what='intensity',
193             axes=False, cmap=None, icut=None, colorbar=False, 
194             output=None, dpi=300):
195        '''
196        Show FFT object = Fourier transform of an image.
197
198        Parameters
199        ----------
200        what : str, optional, default is 'intensity'
201            What result should be shown - 'intensity' or 'phase'.
202        axes : bool, optional, default is False
203            Show axes around the plotted/shown image.
204        cmap : str, optional, default is None
205            Matplotlib cmap name, such as 'magma' or 'viridis'.
206        icut : int or float, optional, default is None
207            Intensity cut value.
208            If icut = 1000, all intenstity values >1000 are set to 1000.
209        colorbar : bool, optional, the default is False
210            If True, a colorbar is added to the plot.
211        output : str or path-like object, optional, default is None
212            If output argument is given, the plot is saved to {output} file.
213        dpi : int, optional, default is 300
214            DPI of the saved image.
215            Relevant only if ouput is not None.
216
217        Returns
218        -------
219        None
220            The output is the image/plot of the Fourier transform result
221            (intensity or phase), which is shown in the screen
222            or (optionally) saved to an image file.
223            
224        Technical note
225        --------------
226        The FFT results (intensity or phase) are shown/plotted using
227        matplotlib. Therefore, many arguments of the current *show* method
228        correspond to matplotlib parameters (such as *cmap* argument).
229        '''
230        
231        # (0) Local function that adds colorbar not exceeding plot height
232        # * this is surprisingly tricky => solution found in StackOverflow
233        #   https://stackoverflow.com/q/18195758
234        # * the usage of the function is slightly non-standard
235        #   see the link above + search for: I created an IPython notebook
236        def add_colorbar(im, aspect=20, pad_fraction=0.5, **kwargs):
237            '''Add a vertical color bar to an image plot.'''
238            divider = axes_grid1.make_axes_locatable(im.axes)
239            width = axes_grid1.axes_size.AxesY(im.axes, aspect=1./aspect)
240            pad = axes_grid1.axes_size.Fraction(pad_fraction, width)
241            current_ax = plt.gca()
242            cax = divider.append_axes("right", size=width, pad=pad)
243            plt.sca(current_ax)
244            return im.axes.figure.colorbar(im, cax=cax, **kwargs)
245        
246        # (1) Determine what to save
247        # (default is FFT.intensity, but FFT.phase can be saved as well
248        if what == 'intensity': arr = self.intensity
249        else: arr = self.phase
250        
251        # (2) Set default colormap
252        if cmap is None: cmap = 'gray'
253        
254        # (3) Plot 
255        # Basic plot, saved to im - this is needed for (optional) colorbar
256        im = plt.imshow(arr, cmap=cmap, vmax=icut)
257        # Add nice colorbar not exceeding image height (using local function)
258        if colorbar: add_colorbar(im)
259 
260        # (4) If saving to output file is required,
261        # remove axes + edges and save the figure.
262        if output is not None:
263            plt.axis('off')
264            plt.savefig(output, dpi=dpi, bbox_inches='tight', pad_inches=0)
265            
266        # Show the figure.
267        if axes == True: plt.axis('on')
268        else: plt.axis('off') 
269        plt.tight_layout()
270        plt.show()

273    def save(self, output='fft.png', what='intensity', 
274             itype='16bit', icut=None, dpi=300):
275        '''
276        Save FFT object = Fourier transform of an image.
277
278        Parameters
279        ----------
280        output : str or path-like object, optional, default is 'fft.png' 
281            Name of the output file.
282        what : str, optional, default is 'intensity'
283            What result should be shown - 'intensity' or 'phase'.
284        itype : str, optional, default is '16bit'
285            Format of the output image.
286            If '16bit' (default) => 16-bit grayscale image.
287            If '8bit' (less suitable, narrow range) => 8-bit grayscale image.
288        icut : int or float, optional, default is None
289            Intensity cut value.
290            If icut = 1000, all intenstity values >1000 are set to 1000.            
291        dpi : int, optional, default is 300
292            DPI of the saved image.
293
294        Returns
295        -------
296        None
297            The output is the image the Fourier transform result
298            (intensity or phase), which is saved in {output} file.
299        
300        Technical note
301        --------------
302        The FFT results (images of 'intensity' or 'phase') can be saved
303        either as matplotlib plots (show method with optional output argument)
304        or standard grayscale images (save method = this method).
305        The save method gives standard result, the show method can yield
306        colour images with various cmaps and/or colorbars,
307        which are suitable for presentations.
308        '''
309        
310        # (1) Determine what to save
311        # (default is FFT.intensity, but FFT.phase can be saved as well        
312        if what == 'intensity': arr = self.intensity
313        else: arr = self.phase
314        
315        # (2) Cut intenstity if required
316        if icut is not None:
317            arr = np.where(arr > icut, icut, arr)
318            
319        # (3) Rescale to 8bit or 16bit image
320        if itype == '16bit': 
321            norm_constant = 2**16 - 1 
322            arr = arr/np.max(arr) * norm_constant
323            arr = arr.astype(np.uint16)
324        else: 
325            norm_constant = 2**8 - 1
326            arr = arr/np.max(arr) * norm_constant
327            arr = arr.astype(np.uint8)
328        
329        # (4) Save array to Image using PIL
330        # (saving with PIL = original number of points/pixels + selected dpi
331        img = Image.fromarray(arr)
332        img.save(output, dpi=(dpi,dpi))
